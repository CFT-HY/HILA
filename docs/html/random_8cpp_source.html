166double hila::gaussrand2(double &out2) {
167
168 double phi, urnd, r;
169 phi = 2.0 * M_PI * hila::random();
170
171 // this should not really trigger
172 do {
173 urnd = 1.0 - hila::random();
174 } while (urnd == 0.0);
175
176 r = sqrt(-::log(urnd) * (2.0 * VARIANCE));
177 out2 = r * cos(phi);
178 return r * sin(phi);
179}
