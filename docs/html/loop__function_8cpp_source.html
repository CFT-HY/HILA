411  std::vector<var_info *> *out_variables,
412  std::vector<var_info *> *dep_variables,
413  argument_info &ai) {
414 
415  // keep filling argument info
416  ai.E = E;
417  ai.PV = pv;
418 
419  // check parameter type
420 
421  // check if we have out_only qualifier

423

521 bool GeneralVisitor::attach_dependent_vars(std::vector<var_info *> &variables, bool sitedep,
522  std::vector<var_info *> &dep_variables) {
523 
524  // if sitedep == true or one of the dep_variables is sitedep, we can mark
525  // all vars in "variables" as site dep.
526  if (!sitedep) {
527  for (auto d : dep_variables) {
528  if (d->is_site_dependent) {
529  sitedep = true;
530  break;
531  }
532  }
533  }
534  if (sitedep) {
535  for (auto v : variables)
536  v->is_site_dependent = true;
537 
538  } else {
539  for (auto v : variables) {
540 
541  // if not site dep. seen so far, variables appearing in args may bring it
542  // later. add dep var, if not there already
543  for (auto d : dep_variables) {
544  bool found = false;
545  for (auto vdep : v->dependent_vars) {
546  if (d == vdep) {
547  found = true;
548  break;
549  }
550  }
551  if (!found)
552  v->dependent_vars.push_back(d);
553  }
554  }
555  }
556  return sitedep;
557 }

573  return true;
