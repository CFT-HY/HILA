6/* Kennedy-Pendleton quasi heat bath on SU(2) subgroups */

8/* This does the heatbath with
9 * beta ( 1 - 1/Ncol Tr U S^+ ) S is the sum of staples.
10 * -> -beta/Ncol Tr u (US+) = -beta/Ncol u_ab (US+)_ba
11 * here u is a SU(2) embedded in SU(N); u = su2 * 1.
12 */

18template <typename T,int N>
19double suN_heatbath(SU<N, T> &U, const SU<N, T> &staple, double beta) {
20 // K-P quasi-heat bath by SU(2) subgroups
21
22 double utry, uhit;
23 double xr1, xr2, xr3, xr4;
24
25 double r, r2, rho, z;
26 double al, d, xl, xd;
27 int k, test;
28 double b3;

30 SU<2,T> h2x2;
31 SU2<T> v, a, h;
32 double pi2 = M_PI * 2.0;
33
34 b3 = beta / N;
35
36 utry = uhit = 0.0;
37
38 /* now for the qhb updating */
39
40 action = U * staple.dagger();
41
42 /* SU(2) subgroups */
43 for (int ina = 0; ina < N - 1; ina++)
44 for (int inb = ina + 1; inb < N; inb++) {
45
46 /* decompose the action into SU(2) subgroups using
47 * Pauli matrix expansion
48 * The SU(2) hit matrix is represented as
49 * a0 + i * Sum j (sigma j * aj)
50 * Let us actually store dagger, v = action(a,b).dagger
51 */
52 v.d = action.e(ina, ina).re + action.e(inb, inb).re;
53 v.c = -(action.e(ina, ina).im - action.e(inb, inb).im);
54 v.a = -(action.e(ina, inb).im + action.e(inb, ina).im);
55 v.b = -(action.e(ina, inb).re - action.e(inb, ina).re);
56
57 z = sqrt(v.det());
58
59 v /= z;
60
61 /* end norm check--trial SU(2) matrix is a0 + i a(j)sigma(j)*/
62 /* test */
63 /* now begin qhb */
64 /* get four random numbers */
65
66 xr1 = log(1.0 - hila::random());
67 xr2 = log(1.0 - hila::random());
68 xr3 = hila::random();
69 xr4 = hila::random();
70
71 xr3 = cos(pi2 * xr3);
72
73
74 // generate a0 component of suN matrix
75 //
76 // first consider generating an su(2) matrix h
77 // according to exp(bg/Ncol * re tr(h*s))
78 // rewrite re tr(h*s) as re tr(h*v)z where v is
79 // an su(2) matrix and z is a real normalization constant
80 // let v = z*v. (z is 2*xi in k-p notation)
81 // v is represented in the form v(0) + i*sig*v (sig are pauli)
82 // v(0) and vector v are real
83 //
84 // let a = h*v and now generate a
85 // rewrite beta/3 * re tr(h*v) * z as al*a0
86 // a0 has prob(a0) = n0 * sqrt(1 - a0**2) * exp(al * a0)
87
88
89 al = b3 * z;
90
91 // let a0 = 1 - del**2
92 // get d = del**2
93 // such that prob2(del) = n1 * del**2 * exp(-al*del**2)
94
95
96 d = -(xr2 + xr1 * xr3 * xr3) / al;
97
98 /* monte carlo prob1(del) = n2 * sqrt(1 - 0.5*del**2)
99 * then prob(a0) = n3 * prob1(a0)*prob2(a0)
100 */
101
102 if ((1.00 - 0.5 * d) <= xr4 * xr4) {
103 if (al > 2.0) { /* k-p algorithm */
104 test = 0;
105 for (k = 0; k < 40 && !test; k++) {
106 /* get four random numbers */
107 xr1 = log(1.0 - hila::random());
108 xr2 = log(1.0 - hila::random());
109 xr3 = hila::random();
110 xr4 = hila::random();
111
112 xr3 = cos(pi2 * xr3);
113
114 d = -(xr2 + xr1 * xr3 * xr3) / al;
115 if ((1.00 - 0.5 * d) > xr4 * xr4)
116 test = 1;
117 }
118 utry += k;
119 uhit++;
120#ifndef __CUDA_ARCH__
121 // TODO: Make parallel-warning out of this
122 // hila::out0 << "site took 20 kp hits\n";
123#endif
124 } else { /* now al <= 2.0 */
125
126 /* creutz algorithm */
127 xl = exp((double)(-2.0 * al));
128 xd = 1.0 - xl;
129 test = 0;
130 double a0;
131 for (k = 0; k < 40 && test == 0; k++) {
132 /* get two random numbers */
133 xr1 = hila::random();
134 xr2 = hila::random();
135
136 r = xl + xd * xr1;
137 a0 = 1.00 + log((double)r) / al;
138 if ((1.0 - a0 * a0) > xr2 * xr2)
139 test = 1;
140 }
141 d = 1.0 - a0;
142 utry += k;
143 uhit++;
144#ifndef __CUDA_ARCH__
145 // hila::out0 << "site took 20 creutz hits\n";
146#endif
147 } /* endif al */
148
149 } else {
150 /* direct hit */
151 utry += 1.0;
152 uhit += 1.0;
153 }
154
155 /* generate full su(2) matrix and update link matrix*/
156
157 /* find a0 = 1 - d*/
158 a.d = 1.0 - d;
159 /* compute r */
160 r2 = 1.0 - a.d * a.d;
161 r2 = fabs(r2);
162 r = sqrt(r2);
163
164 /* compute a3 */
165 a.c = (2.0 * hila::random() - 1.0) * r;
166
167 /* compute a1 and a2 */
168 rho = r2 - a.c * a.c;
169 rho = sqrt(fabs(rho));
170
171 /*xr2 is a random number between 0 and 2*pi */
172 xr2 = pi2 * hila::random();
173 a.a = rho * cos((double)xr2);
174 a.b = rho * sin((double)xr2);
175
176 /* now do the updating. h = a*v^dagger, new u = h*u */
177 h = a * v; // v was daggerized above
178
179 /* Elements of SU(2) matrix */
180 h2x2 = h.convert_to_2x2_matrix();
181
182 /* update the link and 'action' */
183
184 U.mult_by_2x2_left(ina,inb,h2x2);
185 action.mult_by_2x2_left(ina,inb,h2x2);
186
187 } /* hits */
188
189 /* check_unitarity( U );
190 */
191 return (uhit / utry);
192
193} /* site */
