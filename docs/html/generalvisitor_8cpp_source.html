451 var_info *GeneralVisitor::handle_var_ref(DeclRefExpr *DRE, bool is_assign,
452  const std::string &assignop, Stmt *assign_stmt,
453  bool is_raw) {
454 
455  if (isa<VarDecl>(DRE->getDecl())) {
456  auto decl = dyn_cast<VarDecl>(DRE->getDecl());
457 
459  clang::QualType typ = decl->getType().getUnqualifiedType().getNonReferenceType();
460  typ.removeLocalConst();
461 
462  // if (typ.getAsString(PP) == "lattice_struct *") llvm::errs() << "GOT
463  // LATTICE_STRUCT PTR!!!\n";
464 
465  if (typ.getAsString(PP) == "X_index_type" || typ.getAsString(PP) == "lattice_struct")
466  return nullptr;
467 
468  if (typ.getAsString(PP) == "SiteSelect") {
469  reportDiag(DiagnosticsEngine::Level::Error, DRE->getSourceRange().getBegin(),
470  "SiteSelect must be referred only via .select() -method inside onsites() "
471  "-loops");
472  }
473 
474  var_ref vr;
475  vr.ref = DRE;
476  // vr.ind = writeBuf->markExpr(DRE);
477  vr.is_assigned = is_assign;
478  if (is_assign)
479  vr.assignop = assignop;
480 
481  bool foundvar = false;
482  var_info *vip = nullptr;
483  for (var_info &vi : var_info_list) {
484  if (vi.decl == decl && !is_raw) {
485  // found already referred to decl
486  // check if this particular ref has been handled before
487  bool foundref = false;
488  for (auto &r : vi.refs)
489  if (r.ref == DRE) {
490  foundref = true;
491  // if old check was not assignment and this is, change status
492  // can happen if var ref is a function "out" argument
493  if (r.is_assigned == false && is_assign == true) {
494  r.is_assigned = true;
495  r.assignop = assignop;
496  }
497  break;
498  }
499  if (!foundref) {
500  // a new reference
501  vi.refs.push_back(vr);
502  }
503  vi.is_assigned |= is_assign;
504  if (is_top_level && vi.reduction_type == reduction::NONE) {
505  vi.reduction_type = get_reduction_type(is_assign, assignop, vi);
506  }
507  vip = &vi;
508  foundvar = true;
509  break;
510 
511  } else if (vi.decl == decl) {
512  // now is_raw == true, old variable, do not have to check refs
513  foundvar = true;
514  break;
515  }
516  }
517  if (!foundvar) {
518  // new variable referred to
519  vip = new_var_info(decl);
520 
521  vip->refs.push_back(vr);
522  vip->is_assigned = is_assign;
523 
524  // llvm::errs() << "New var info " << vip->name << " assign " << vip->is_assigned << "
525  // loop local " << vip->is_loop_local << " reduction " << (vip->reduction_type !=
526  // reduction::NONE) << '\n';
527 
528  if (is_raw) {
529  vip->is_raw = true;
530  vip->is_site_dependent = true; // probably does not matter
531  } else {
532 
533  // we know refs contains only 1 element
534  if (is_top_level) {
535  vip->reduction_type = get_reduction_type(is_assign, assignop, *vip);
536  }
537  }
538  }
539 
540  if (!is_raw && is_assign && assign_stmt != nullptr && !vip->is_site_dependent) {
541  vip->is_site_dependent = is_rhs_site_dependent(assign_stmt, &vip->dependent_vars);
542 
543  // llvm::errs() << "Var " << vip->name << " depends on site: " <<
544  // vip->is_site_dependent << "\n";
545  }
546  return vip;
547 
548  } else {
549  // end of VarDecl - how about other decls, e.g. functions?
550  reportDiag(DiagnosticsEngine::Level::Error, DRE->getSourceRange().getBegin(),
551  "reference to unimplemented (non-variable) type");
552  }
553 
554  return nullptr;
555 }
