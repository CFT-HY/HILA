40 // use here function instead of method, works for double/float too

156 Algebra<SU<N, T>> a;
157
158 // diagonal generators
159 T sum = this->e(0, 0).im;
160 for (int i = 1; i < N; i++) {
161 a.e(i - 1) = (sum - i * this->e(i, i).im) / sqrt(0.5 * i * (i + 1));
162 sum += this->e(i, i).im;
163 }
164
165 // Then off-diag bits
166 int k = a.n_diag;
167 for (int i = 0; i < N - 1; i++) {
168 for (int j = i + 1; j < N; j++) {
169 auto od = this->e(i, j) - this->e(j, i).conj();
170 a.e(k) = od.re;
171 a.e(k + 1) = od.im;
172 k += 2;
173 }

205 // suN generators, normalized as
206 // Tr(\lambda_i\lambda_j) = 1/2 \delta_ij
207 // off-diagonal are just that:
208 // \lambda^od_ij,r = 1/2 for elements ij and ji
209 // \lambda^od_ij,i = i/2 for ij; -i for ji
210 //
211 // diagonals: su(N) has N-1 diag generators
212 // parametrize these recursively:
213 // \lambda_1 = diag(1,-1,0,0,..)/sqrt(1)/2
214 // \lambda_2 = diag(1,1,-2,0,..)/sqrt(3)/2
215 // \lambda_3 = diag(1,1,1,-3,..)/sqrt(6)/2
216 // \lambda_i = diag(1,.. ,-i,..)/sqrt(i(i+1)/2)/2
217 // ..
218 // \lambda_N-1 = diag(1,.. ,1,-(N-1))/sqrt(N(N-1)/2)/2
219 //
220 // Define \lambda's so that diagonals come first

223 SU<N, T> expand() const {
224 SU<N, T> m;
225
226 Vector<N, T> d;
227
228 d.e(0) = (T)0;
229 for (int i = 1; i < N; i++) {
230 T r = this->e(i - 1) * sqrt(0.5 / (i * (i + 1)));
231 // the contributions from 1's above
232 for (int j = 0; j < i; j++)
233 d.e(j) += r;
234 // and set the negative element - no sum here, first contrib
235 d.e(i) = -i * r;
236 }
237
238 for (int i = 0; i < N; i++)
239 m.e(i, i) = Complex<T>(0, d.e(i));
240
241 int k = n_diag;
242 T inv2 = 1.0 / 2;
243 for (int i = 0; i < N - 1; i++)
244 for (int j = i + 1; j < N; j++) {
245 Complex<T> v(this->c[k] * inv2, this->c[k + 1] * inv2);
246 m.e(i, j) = v;
247 m.e(j, i) = -v.conj();
248 k += 2;
249 }
250
251 return m;
252 }
