231 void reduce_sum_node(const T &v) {
232
233 wait(); // wait for possible ongoing
234
235 // add the node values to reduction var
236 if (hila::myrank() == 0 || is_delayed_)
237 val += v;
238 else
239 val = v;
240
241 if (is_delayed_) {
242 if (delay_is_on && is_delayed_sum == false) {
243 assert(0 && "Cannot mix sum and product reductions!");
244 }
245 delay_is_on = true;
246 is_delayed_sum = true;
247 } else {
248 do_reduce_operation(MPI_SUM);
249 }
250 }
251
256 void reduce_product_node(const T &v) {
257
258 wait();
259
260 if (hila::myrank() == 0 || is_delayed_)
261 val *= v;
262 else
263 val = v;
264
265 if (is_delayed_) {
266 if (delay_is_on && is_delayed_sum == true) {
267 assert(0 && "Cannot mix sum and product reductions!");
268 }
269 delay_is_on = true;
270 is_delayed_sum = false;
271 } else {
272 do_reduce_operation(MPI_PROD);
273 }

339template <typename T>
340T Field<T>::minmax(bool is_min, Parity par, CoordinateVector &loc) const {
341
342 static_assert(
343 std::is_same<T, int>::value || std::is_same<T, long>::value ||
344 std::is_same<T, float>::value || std::is_same<T, double>::value ||
345 std::is_same<T, long double>::value,
346 "In Field .min() and .max() methods the Field element type must be one of "
347 "(int/long/float/double/long double)");
348
349#if defined(CUDA) || defined(HIP)
350 T val = gpu_minmax(is_min, par, loc);
351#else
352 int sgn = is_min ? 1 : -1;
353 // get suitable initial value
354 T val = is_min ? std::numeric_limits<T>::max() : std::numeric_limits<T>::min();
355
356// write the loop with explicit OpenMP parallel region. It has negligible effect
357// on non-OpenMP code, and the pragmas are ignored.
358#pragma omp parallel shared(val, loc, sgn, is_min)
359 {
360 CoordinateVector loc_th(0);
361 T val_th =
362 is_min ? std::numeric_limits<T>::max() : std::numeric_limits<T>::min();
363
364// Pragma "hila omp_parallel_region" is necessary here, because this is within
365// omp parallel
366#pragma hila novector omp_parallel_region direct_access(loc_th, val_th)
367 onsites (par) {
368 if (sgn * (*this)[X] < sgn * val_th) {
369 val_th = (*this)[X];
370 loc_th = X.coordinates();
371 }
372 }
373
374#pragma omp critical
375 if (sgn * val_th < sgn * val) {
376 val = val_th;
377 loc = loc_th;
378 }
379 }
380#endif
381
382 if (hila::number_of_nodes() > 1) {
383 size_t size;
384 MPI_Datatype dtype = get_MPI_number_type<T>(size, true);
385
386 struct {
387 T v;
388 int rank;
389 } rdata;
390
391 rdata.v = val;
392 rdata.rank = hila::myrank();
393
394 // after allreduce rdata contains the min value and rank where it is
395 if (is_min) {
396 MPI_Allreduce(MPI_IN_PLACE, &rdata, 1, dtype, MPI_MINLOC,
397 lattice.mpi_comm_lat);
398 } else {
399 MPI_Allreduce(MPI_IN_PLACE, &rdata, 1, dtype, MPI_MAXLOC,
400 lattice.mpi_comm_lat);
401 }
402 val = rdata.v;
403
404 // send the coordinatevector of the minloc to all nodes
405 MPI_Bcast(&loc, sizeof(CoordinateVector), MPI_BYTE, rdata.rank,
406 lattice.mpi_comm_lat);
407 }
408
409 return val;
410}
