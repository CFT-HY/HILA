52

54 const char *p = argv[i];

67 const char *get_cstring(const char *flag) {
68 int flaglen = strlen(flag);
69 assert(flaglen > 0);
70
71 for (int i = 1; i < argc; i++) {
72 const char *p = argv[i];
73
74 // OK if p starts with flag
75 if (std::strcmp(p, flag) == 0) {
76 if (i > argc - 2) {
77 hila::out0 << "Expecting an argument after command line parameter '" << flag
78 << "'\n";
79 hila::terminate(0);
80 }
81 p = argv[i + 1];
82 argc -= 2;
83 for (; i < argc; i++)
84 argv[i] = argv[i + 2];
85 return (p);
86 }
87 }
88 return (nullptr);
89 }

91 long get_int(const char *flag) {
92 const char *p = get_cstring(flag);
93 char *end;
94
95 if (p == nullptr)
96 return LONG_MAX; // not found
97
98 long val = strtol(p, &end, 10);
99 if ((errno == ERANGE && (val == LONG_MAX || val == LONG_MIN)) || (errno != 0 && val == 0) ||
100 end == p || *end != 0) {
101 hila::out0 << "Expect a number (integer) after command line parameter '" << flag
102 << "'\n";
103 hila::terminate(0);
104 }
105 return val;
106 }

479void setup_partitions(cmdlineargs &commandline) {
480
481 // get partitions cmdlinearg first
482 long lnum = commandline.get_int("-partitions");
483 if (lnum <= 0) {
484 hila::out0 << "partitions=<number> command line argument value must be positive "
485 "integer (or argument omitted)\n";
486 hila::finishrun();
487 }
488 if (lnum == LONG_MAX) {
489 hila::partitions._number = 1;
490 } else {
491 hila::partitions._number = lnum;
492 }
493
494 if (hila::partitions.number() == 1)
495 return;
496
497 hila::out0 << " Dividing nodes into " << hila::partitions.number() << " partitions\n";
498
499 if (hila::number_of_nodes() % hila::partitions.number()) {
500 hila::out0 << "** " << hila::number_of_nodes() << " nodes not evenly divisible into "
501 << hila::partitions.number() << " partitions\n";
502 hila::finishrun();
503 }
504
505#if defined(BLUEGENE_LAYOUT)
506 hila::partitions._mylattice = bg_layout_partitions(hila::partitions.number());
507#else // generic
508 hila::partitions._mylattice =
509 (hila::myrank() * hila::partitions.number()) / hila::number_of_nodes();
510 /* and divide system into partitions */
511 if (!hila::check_input)
512 split_into_partitions(hila::partitions.mylattice());
513#endif
514
515 const char *p = commandline.get_cstring("-o");
516 std::string fname;
517 if (p != nullptr)
518 fname = p + std::to_string(hila::partitions.mylattice());
519 else
520 fname = DEFAULT_OUTPUT_NAME + std::to_string(hila::partitions.mylattice());
521
522 // now need to open output file
523
524 hila::out.flush(); // this should be cout at this stage
525
526 // all nodes open the file -- perhaps not? Leave only node 0
527 int do_exit = 0;
528 if (hila::myrank() == 0 && !hila::check_input) {
529 hila::output_file.open(fname, std::ios::out | std::ios::app);
530 if (hila::output_file.fail()) {
531 std::cout << "Cannot open output file " << fname << '\n';
532 do_exit = 1;
533 }
534 }
535
536 hila::broadcast(do_exit);
537
538 if (do_exit)
539 hila::finishrun();
540
541 hila::out.flush();
542 if (!hila::check_input) {
543 hila::out.rdbuf(hila::output_file.rdbuf());
544 // output now points to output_redirect
545 if (hila::myrank() == 0) {
546 hila::out0.rdbuf(hila::out.rdbuf());
547 }
548 }
549 hila::out0 << " ---- SPLIT " << hila::number_of_nodes() << " nodes into "
550 << hila::partitions.number() << " partitions, this " << hila::partitions.mylattice()
551 << " ----\n";
552
553
554 /* Default sync is no */

556 hila::partitions._sync = true;
557 hila::out0 << "Synchronising partition trajectories\n";
558 } else {
