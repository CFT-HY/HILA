25inline Vector<NDIM, double> convert_to_k(const CoordinateVector &cv) {
26 Vector<NDIM, double> k;
27 foralldir (d) {
28 int n = pmod(cv.e(d), lattice.size(d));
29 if (n > lattice.size(d) / 2)
30 n -= lattice.size(d);

175 template <typename T>
176 void collect_data(const Field<T> &f) {
177
178 extern hila::timer pencil_collect_timer;
179 pencil_collect_timer.start();
180
181 constexpr int elements = sizeof(T) / sizeof(cmplx_t);
182
183 // Build vector offset, which encodes where the data should be written
184 // elem_offset is the same for the offset of the elements of T
185 CoordinateVector offset, nmin;
186
187 const size_t elem_offset =
188 pencil_get_buffer_offsets(dir, sizeof(T) / sizeof(cmplx_t), offset, nmin);
189
190 cmplx_t *sb = send_buf;
191
192 // and collect the data
193#pragma hila novector direct_access(sb)
194 onsites(ALL) {
195
196 T_union<T, cmplx_t> v;
197 v.val = f[X];
198 int off = offset.dot(X.coordinates() - nmin);
199 for (int i = 0; i < elements; i++) {
200 sb[off + i * elem_offset] = v.c[i];
201 }
202 }
203
204 pencil_collect_timer.stop();
205 }

212 constexpr int elements = sizeof(T) / sizeof(cmplx_t);

245 void reshuffle_data(Direction prev_dir) {
246
247 extern hila::timer pencil_reshuffle_timer;
248 pencil_reshuffle_timer.start();
249
250 int elem = elements;
251
252 CoordinateVector offset_in, offset_out, nmin;
253
254 const size_t e_offset_in =
255 pencil_get_buffer_offsets(prev_dir, elements, offset_in, nmin);
256 const size_t e_offset_out =
257 pencil_get_buffer_offsets(dir, elements, offset_out, nmin);
258
259 cmplx_t *sb = send_buf;
260 cmplx_t *rb = receive_buf;
261
262#pragma hila novector direct_access(sb, rb)
263 onsites(ALL) {
264 CoordinateVector v = X.coordinates() - nmin;
265 size_t off_in = offset_in.dot(v);
266 size_t off_out = offset_out.dot(v);
267 for (int e = 0; e < elem; e++) {
268 sb[off_out + e * e_offset_out] = rb[off_in + e * e_offset_in];
269 }
270 }
271
272 pencil_reshuffle_timer.stop();
273 }

496template <typename T>
497Field<hila::scalar_type<T>> Field<T>::FFT_complex_to_real(fft_direction fftdir) const {
498
499 static_assert(
500 hila::is_complex<T>::value,
501 "FFT_complex_to_real can be applied only to Field<Complex<>> type variable");
502
503 // first, do a full reflection of the field, giving rf(x) = f(L-x) = "f(-x)"
504 auto rf = this->reflect();
505 // And symmetrize the field appropriately - can use rf
506 onsites(ALL) {
507 int type = FFT_complex_to_real_loc(X.coordinates());
508 if (type == 1) {
509 rf[X] = (*this)[X];
510 } else if (type == -1) {
511 rf[X] = rf[X].conj();
512 } else {
513 rf[X].real() = (*this)[X].real();
514 rf[X].imag() = 0;
515 }
516 }
517
518 FFT_field(rf, rf, fftdir);
519
520 double ims = 0;
521 double rss = 0;
522 onsites(ALL) {
523 ims += ::squarenorm(rf[X].imag());
524 rss += ::squarenorm(rf[X].real());
525 }
526
527 Field<hila::scalar_type<T>> res;
528 onsites(ALL) res[X] = rf[X].real();
529 return res;
530}
