69/* Set seed on device and host */
70void hila::initialize_device_rng(uint64_t seed) {
71 unsigned long n_blocks = (lattice.mynode.volume() + N_threads - 1)/ N_threads;
72
73#if defined(GPU_RNG_THREAD_BLOCKS) && GPU_RNG_THREAD_BLOCKS > 0
74 // If we have limited rng block number
75 if (GPU_RNG_THREAD_BLOCKS < n_blocks) {
76 n_blocks = GPU_RNG_THREAD_BLOCKS;
77 }
78
79 hila::out0 << "GPU random number generator initialized\n";
80 hila::out0 << "GPU random number thread blocks: " << n_blocks << " of size " << N_threads << " threads\n";
81#elif defined(GPU_RNG_THREAD_BLOCKS) && GPU_RNG_THREAD_BLOCKS < 0
82 hila::out0 << "GPU RANDOM NUMBERS DISABLED, GPU_RNG_THREAD_BLOCKS < 0\n";
83#else
84 hila::out0 << "GPU random number generator initialized\n";
85 hila::out0 << "GPU random numbers: using on generator/site (GPU_RNG_THREAD_BLOCKS = 0 or undefined)\n";
86#endif
87
88 unsigned long long n_sites = n_blocks * N_threads;
89 unsigned long long myseed = seed + hila::myrank() * n_sites;
90
91 // allocate random state and copy the ptr to d_gpurandstateptr
92 gpuMalloc(&gpurandstateptr, n_sites * sizeof(gpurandState));
93 gpuMemcpyToSymbol(d_gpurandstateptr, &gpurandstateptr, sizeof(gpurandState *), 0,
94 gpuMemcpyHostToDevice);
95
96#ifdef CUDA
97 seed_random_kernel<<<n_blocks, N_threads>>>(myseed);
98#else
99 hipLaunchKernelGGL(seed_random_kernel, dim3(n_blocks), dim3(N_threads), 0, 0, myseed);
100#endif
101 check_device_error("seed_random kernel");
102}

154__device__ const CoordinateVector backend_lattice_struct::coordinates(unsigned idx) const {
155 CoordinateVector c;
156 unsigned vdiv, ndiv;
157
158 vdiv = idx;
159 for (int d = 0; d < NDIM - 1; ++d) {
160 ndiv = vdiv / _d_nodesize[d];
161 c[d] = vdiv - ndiv * _d_nodesize[d] + _d_nodemin[d];
162 vdiv = ndiv;
163 }
164 c[NDIM - 1] = vdiv + _d_nodemin[NDIM - 1];
165
166 return c;
167}

176void backend_lattice_struct::setup(lattice_struct &lattice) {
177 CoordinateVector *tmp;
178
179 /* Setup neighbour fields in all directions */
180 for (int d = 0; d < NDIRS; d++) {
181 // For normal boundaries
182 gpuMalloc(&(d_neighb[d]), lattice.mynode.volume() * sizeof(unsigned));
183 gpuMemcpy(d_neighb[d], lattice.neighb[d], lattice.mynode.volume() * sizeof(unsigned),
184 gpuMemcpyHostToDevice);
185
186#ifdef SPECIAL_BOUNDARY_CONDITIONS
187 // For special boundaries
188 // TODO: check this really works now!
189 const unsigned *special_neighb =
190 lattice.get_neighbour_array((Direction)d, hila::bc::ANTIPERIODIC);
191
192 if (special_neighb != lattice.neighb[d]) {
193 gpuMalloc(&(d_neighb_special[d]), lattice.mynode.volume() * sizeof(unsigned));
194 gpuMemcpy(d_neighb_special[d], special_neighb,
195 lattice.mynode.volume() * sizeof(unsigned), gpuMemcpyHostToDevice);
196 } else {
197 d_neighb_special[d] = d_neighb[d];
198 }
199#endif
200 }
201
202#ifdef EVEN_SITES_FIRST
203 /* Setup the location field */
204 gpuMalloc(&(d_coordinates), lattice.mynode.volume() * sizeof(CoordinateVector));
205 tmp = (CoordinateVector *)memalloc(lattice.mynode.volume() * sizeof(CoordinateVector));
206 for (unsigned i = 0; i < lattice.mynode.volume(); i++)
207 tmp[i] = lattice.coordinates(i);
208
209 gpuMemcpy(d_coordinates, tmp, lattice.mynode.volume() * sizeof(CoordinateVector),
210 gpuMemcpyHostToDevice);
211 free(tmp);
212#endif
213
214 // Other backend_lattice parameters
215 field_alloc_size = lattice.field_alloc_size();
216
217 int64_t v = lattice.volume();
218 gpuMemcpyToSymbol(_d_volume, &v, sizeof(int64_t), 0, gpuMemcpyHostToDevice);
219 int s[NDIM];
220 foralldir(d) s[d] = lattice.size(d);
221 gpuMemcpyToSymbol(_d_size, s, sizeof(int) * NDIM, 0, gpuMemcpyHostToDevice);
222
223#ifndef EVEN_SITES_FIRST
224 foralldir(d) s[d] = lattice.mynode.size[d];
225 gpuMemcpyToSymbol(_d_nodesize, s, sizeof(int) * NDIM, 0, gpuMemcpyHostToDevice);
226
227 foralldir(d) s[d] = lattice.mynode.min[d];
228 gpuMemcpyToSymbol(_d_nodemin, s, sizeof(int) * NDIM, 0, gpuMemcpyHostToDevice);
229
230 foralldir(d) s[d] = lattice.mynode.size_factor[d];
231 gpuMemcpyToSymbol(_d_nodefactor, s, sizeof(int) * NDIM, 0, gpuMemcpyHostToDevice);
232
233#endif
234}

236void initialize_gpu(int rank) {
237 int n_devices, my_device;
238
239 gpuGetDeviceCount(&n_devices);
240 check_device_error("Could not get device count");
241 // This assumes that each node has the same number of mpi ranks and GPUs
242 my_device = rank % n_devices;
243
244 hila::out0 << "GPU devices accessible from node 0: " << n_devices << '\n';
245
246 // TODO: this only for node 0?
247 if (n_devices > 1 && rank < 6) {
248 hila::out << "GPU: MPI rank " << rank << " choosing device " << my_device << '\n';
249 if (hila::number_of_nodes() > 6) {
250 hila::out0 << " + " << hila::number_of_nodes() - 6 << " more nodes\n";
251 }
252 }
253
254 gpuSetDevice(my_device);
255
256 // set gpu rng state to "off", to prevent accidental use
257 gpurandstateptr = nullptr;
258 // set d_gpurandstateptr <- nullptr.
259 gpuMemcpyToSymbol(d_gpurandstateptr, &gpurandstateptr, sizeof(gpurandState *), 0,
260 gpuMemcpyHostToDevice);
261
262
263#if defined(CUDA_MALLOC_ASYNC)
264 // set memory pool
265 cudaMemPool_t mempool;
266 cudaDeviceGetDefaultMemPool(&mempool, my_device);
267 uint64_t threshold = UINT64_MAX;
268 cudaMemPoolSetAttribute(mempool, cudaMemPoolAttrReleaseThreshold, &threshold);
269
270#endif
271}
