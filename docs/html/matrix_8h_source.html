745 template <int N>
746 Mtype sort(Vector<N, int> &permutation, hila::sort order = hila::sort::ascending) const {
747
748 static_assert(n == 1 || m == 1, "Sorting possible only for vectors");
749 static_assert(hila::is_arithmetic<T>::value,
750 "Sorting possible only for arithmetic vector elements");
751 static_assert(N == Mtype::size(), "Incorrect size of permutation vector");
752
753 for (int i = 0; i < N; i++)
754 permutation[i] = i;
755 if (hila::sort::nonsorted == order) {
756 return *this;
757 }
758
759 if (hila::sort::ascending == order) {
760 for (int i = 1; i < N; i++) {
761 for (int j = i; j > 0 && c[permutation[j - 1]] > c[permutation[j]]; j--)
762 hila::swap(permutation[j], permutation[j - 1]);
763 }
764 } else {
765 for (int i = 1; i < N; i++) {
766 for (int j = i; j > 0 && c[permutation[j - 1]] < c[permutation[j]]; j--)
767 hila::swap(permutation[j], permutation[j - 1]);
768 }
769 }
770
771 return this->reorder(permutation);
772 }

789 "Cannot multiply real matrix with complex 2x2 block matrix");

792 for (int i = 0; i < m; i++) {
793 a.e(0) = this->e(p, i);
794 a.e(1) = this->e(q, i);
795

799 this->e(q, i) = a.e(1);

825#pragma hila novector
826 template <typename Et, typename Mt, typename MT,
827 typename Dtype = typename std::conditional<hila::contains_complex<T>::value,
828 Complex<double>, double>::type>
829 int eigen_jacobi(out_only Vector<n, Et> &eigenvaluevec,
830 out_only Matrix_t<n, n, Mt, MT> &eigenvectors,
831 enum hila::sort sorted = hila::sort::nonsorted) const {
832
833 static_assert(!hila::contains_complex<T>::value || hila::contains_complex<Mt>::value,
834 "Eigenvector matrix must be complex with complex original matrix");
835
836 static_assert(n == m, "Eigensystem can be solved only for square matrices");
837 int rot;
838 SquareMatrix<n, Dtype> M, V;
839 Vector<n, double> eigenvalues;
840
841 // Do it in double prec; copy fields
842 V = 1;
843 M = (*this);
844
845 // don't need the imag. parts of diag (are zero)
846 eigenvalues = M.diagonal().real();
847
848 for (rot = 0; rot < 300; rot++) {
849
850 /* find the largest element */
851 int p, q;
852 double abs_mpq = -1.0; // this quarantees that even trivial matrix works
853 for (int i = 0; i < n - 1; i++) {
854 for (int j = i + 1; j < n; j++) {
855 double t = ::squarenorm(M.e(i, j));
856 if (abs_mpq < t) {
857 abs_mpq = t;
858 p = i;
859 q = j;
860 }
861 }
862 }
863
864 abs_mpq = ::sqrt(abs_mpq);
865
866 // if off-diag elements are tiny return
867
868 if (abs_mpq <= 1e-18 * (std::abs(eigenvalues[p]) + std::abs(eigenvalues[q]))) {
869 if (sorted == hila::sort::nonsorted) {
870
871 // return values and vectors as is
872 eigenvaluevec = eigenvalues;
873 eigenvectors = V;
874
875 } else {
876 // bubble sort eigenvalues to decreasing order
877 Vector<n, int> perm;
878 eigenvaluevec = eigenvalues.sort(perm, sorted);
879 eigenvectors = V.reorder_columns(perm);
880 }
881 return (rot);
882 }
883
884 Dtype mpq = M.e(p, q);
885
886 /* now do the p,q-rotation:
887 * M <- P^+ M P, where
888 * | c s | p
889 * P = |-s* c* | q = Ppq
890 * | 1 | r
891 *
892 * with P^+ = P^-1 -> cc* + ss* = 1, det P = 1
893 * Thus, P is SU(2) or O(2) matrix
894 *
895 * | c* -s | | mpp mpq mpr| | c s |
896 * M = | s* c | | mqp mqq mqr| |-s* c* |
897 * | 1| | mrp mrq mrr| | 1|
898 *
899 * = Pip* Mij Pjq, mqp = mpq*
900 *
901 * Set now Mpq = (c*mpp - s mqp)s + (c*mpq -smqq)c* = 0
902 * = c*^2 mpq - s^2 mpq* + c*s (mpp - mqq)
903 * = |mpq| [ c*^2 e - s^2 e* + c*s (mpp-mqq)/|mpq| ]
904 * where e = exp(i arg(mpq)) = mpq/|mpq|, e* = 1/e
905 *
906 * Now the "rotation angle" (c~cos\phi, s~sin\phi)
907 * a = "cot2\phi" = c*^2 e - s^2 e* / 2c*s = (mqq - mpp) / 2|mpq|
908 * Def t = s/c*e, so that the above is
909 * t^2 + 2ta - 1 = 0 ->
910 * t = -a +- sqrt(a^2+1). Choose one with smaller |t|!
911 * This is prone to cancellation, ~ a - a, so write it as
912 * t = sgn(a)/(|a| + sqrt(a^2 + 1).
913 * Now can choose real c*
914 * c* = 1/sqrt(t^2+1)
915 * s = t c* e (and c*c + s*s = 1)
916 */
917
918 double a = (eigenvalues[q] - eigenvalues[p]) / (2 * abs_mpq);
919 double t = 1.0 / (std::abs(a) + std::sqrt(a * a + 1.0));
920 if (a < 0.0)
921 t = -t;
922 double c = 1.0 / std::sqrt(t * t + 1.0);
923
924 Dtype s = mpq * (t * c / abs_mpq);
925 Matrix<2, 2, Dtype> P;
926 P.e(0, 0) = P.e(1, 1) = 1.0 / std::sqrt(t * t + 1.0);
927 P.e(0, 1) = s;
928 P.e(1, 0) = -::conj(s);
929
930 M.mult_by_2x2_left(p, q, P.dagger());
931 M.mult_by_2x2_right(p, q, P);
932
933 eigenvalues[p] = ::real(M.e(p, p));
934 eigenvalues[q] = ::real(M.e(q, q));
935
936 // p,q -elements of m should be 0 - set explictly to avoid rounding erros
937 M.e(p, q) = 0;
938 M.e(q, p) = 0;
939
940 /* Now M done, take care of the ev's too ..
941 * V' = V P = |vpp vpq vpr| | c s | = V_ik P_kj
942 * |vqp vqq vqr| |-s* c |
943 * |vrp vrq vrr| | 1|
944 * vip <- vip c - viq s*
945 * viq <- vip s + viq c
946 * vir <- vir
947 */
948
949 V.mult_by_2x2_right(p, q, P);
950 }
951
952 return (-1);
953 }
