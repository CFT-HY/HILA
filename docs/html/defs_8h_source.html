54// This below declares "out_only" -qualifier. It is empty on purpose. Do not remove!
55#define out_only
56// out_only indicates that the function does not use the original value of the argument:
57// double func( out_only double & p, ..) { ... }
58// func(a);
59// This helps hilapp to optimize memory access. It is not a bug to leave out out_only,
60// but it is a bug to "lie", i.e. use the original value of the variable if out_only is
61// used.
62//
63// out_only for class methods tells hilapp that the base variable original value is not
64// needed:
65// class C {
66// int set() out_only { .. }
67// };
68// indicates that a.set(); does not need original value of a. e qualifier.
69
70// Defined empty on purpose, same as above!
71#define const_function
72// const_function does not change the base variable, but can return a (non-const)
73// reference. Needed typically for access operators for loop extern variables:
74// class v {
75// double c[N];
76// double & e(const int i) const_function { return c[i]; }
77// };
78//
79// v vv;
80// Field<v> f;
81// onsites(ALL) { f[X].e(0) += vv.e(0); }
82// This would not work without const_function, because vv.e(0) might modify loop
83// extern variable vv, which is not allowed. If method is marked "const",
84// then the assignment would not work.
85//
86// const_function is weaker than const.
