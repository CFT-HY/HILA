91void timer::report(bool print_not_timed) {
92 if (hila::myrank() == 0) {
93 char line[202];
94
95 // time used during the counter activity
96 double ttime = gettime();
97 if (count > 0 && !is_error) {
98 if (t_total / count > 0.1) {
99 std::snprintf(line, 200, "%-20s: %14.3f %14ld %10.3f s %8.4f\n", label.c_str(),
100 t_total, (long)count, t_total / count, t_total / ttime);
101 } else if (t_total / count > 1e-4) {
102 std::snprintf(line, 200, "%-20s: %14.3f %14ld %10.3f ms %8.4f\n", label.c_str(),
103 t_total, (long)count, 1e3 * t_total / count, t_total / ttime);
104 } else {
105 std::snprintf(line, 200, "%-20s: %14.3f %14ld %10.3f μs %8.4f\n", label.c_str(),
106 t_total, (long)count, 1e6 * t_total / count, t_total / ttime);
107 }
108 hila::out << line;
109 } else if (!is_error && print_not_timed) {
110 std::snprintf(line, 200, "%-20s: no timed calls made\n", label.c_str());
111 hila::out << line;
112 } else if (is_error) {
113 std::snprintf(line, 200, "%-20s: error:unbalanced start/stop\n", label.c_str());
114 hila::out << line;
115 }
116 }
117}

174bool time_to_exit() {
175 static double max_interval = 0.0;
176 static double previous_time = 0.0;
177 bool finish;
178
179 // if no time limit set
180 if (timelimit == 0.0)
181 return false;
182
183 if (hila::myrank() == 0) {
184 double this_time = gettime();
185 if (this_time - previous_time > max_interval)
186 max_interval = this_time - previous_time;
187 previous_time = this_time;
188
189 // Give 5 min margin for the exit - perhaps needed for writing etc.
190 if (timelimit - this_time < max_interval + 5 * 60.0)
191 finish = true;
192 else
193 finish = false;
194
195 hila::out << "TIMECHECK: " << this_time << "s used, " << timelimit - this_time
196 << "s remaining\n";
197 if (finish)
198 hila::out << "CPU TIME LIMIT, EXITING THE PROGRAM\n";
199 }
200 hila::broadcast(finish);
201 return finish;
202}
