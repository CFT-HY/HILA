32template <typename T>
33T get_ch_inv(const T &U) {
34 T tB[2];
35 Complex<hila::scalar_type<T>> tc;
36 int ip, iip;
37 ip = 0;
38 iip = 1;
39 tB[ip] = 1.;
40 tc = trace(U);
41 for (int k = 2; k <= T::size(); ++k) {
42 tB[iip] = U * tB[ip];
43 tB[iip] -= tc;
44 tc = trace(U * tB[iip]) / k;
45 ip = iip;
46 iip = (iip + 1) % 2;
47 }
48 return tB[ip] / tc;
49}

51template <typename T>
52T get_bp_Amat(const T &U) {
53 T tA1;
54 T tA2;
55 tA1 = 0.5 * U;
56 tA1 += 0.5;
57 tA2 = get_ch_inv(tA1);
58 tA1 = tA2 * tA2.dagger();
59 return tA1 * tA1 * tA2;
60}

62template <typename T>
63T get_bp_iOsqmat(const T &U) {
64 T tA1;
65 T tA2;
66 tA1 = 0.5 * U;
67 tA1 += 0.5;
68 tA2 = tA1.dagger() * tA1;
69 tA1 = get_ch_inv(tA2);
70 tA2 = tA1 * tA1;
71 tA2 -= 1.;
72 return tA2;
73}

75template <typename T>
76void plaqpm(const GaugeField<T> &U, Field<T> &plaqp, Direction d1, Direction d2) {
77
78 Field<T> lower;
79
80 if (d2 != d1) {
81
82 // anticipate that these are needed
83 // not really necessary, but may be faster
84 U[d2].start_gather(d1, ALL);
85 U[d1].start_gather(d2, ALL);
86
87 // calculate first lower 'U' of the staple sum
88 // do it on opp parity
89 onsites(ALL) {
90 lower[X] = U[d2][X].dagger() * U[d1][X] * U[d2][X + d1];
91 }
92
93 // calculate then the upper 'n', and add the lower
94 // lower could also be added on a separate loop
95 onsites(ALL) {
96 auto p1 = U[d1][X] * U[d2][X + d1] * (U[d2][X] * U[d1][X + d2]).dagger();
97 auto p2 = U[d1][X] * lower[X - d2].dagger();
98 plaqp[X] = p1 * get_bp_Amat(p1) + p2 * get_bp_Amat(p2);
99 }
100 }
101}

103template <typename T>
104void plaqpm_db(const GaugeField<T> &U, Field<T> &plaqp, Direction d1, Direction d2, double deltab) {
105
106 Field<T> lower;
107
108 if (d2 != d1) {
109
110 // anticipate that these are needed
111 // not really necessary, but may be faster
112 U[d2].start_gather(d1, ALL);
113 U[d1].start_gather(d2, ALL);
114
115 // calculate first lower 'U' of the staple sum
116 // do it on opp parity
117 onsites(ALL) {
118 double m;
119 if (2 * X.z() < lattice.size(e_z))
120 m = 1.0 - deltab;
121 else
122 m = 1.0 + deltab;
123
124 auto p2 = U[d1][X + d2] * (U[d2][X].dagger() * U[d1][X] * U[d2][X + d1]).dagger();
125 lower[X] = m * p2 * get_bp_Amat(p2);
126 }
127
128 // calculate then the upper 'n', and add the lower
129 // lower could also be added on a separate loop
130 onsites(ALL) {
131 double m;
132 if (2 * X.z() < lattice.size(e_z))
133 m = 1.0 - deltab;
134 else
135 m = 1.0 + deltab;
136
137 auto p1 = U[d1][X] * U[d2][X + d1] * (U[d2][X] * U[d1][X + d2]).dagger();
138 plaqp[X] = m * p1 * get_bp_Amat(p1) + lower[X - d2];
139 }
140 }

144template <typename group>
145void update_E_bp(const GaugeField<group> &U, VectorField<Algebra<group>> &E, const parameters &p,
146 double delta) {
147
148 Field<group> plaqp;
149 hila::scalar_type<group> eps = delta * 2.0 * p.beta / group::size();
150
151 foralldir(d1) {
152 foralldir(d2) if (d2 != d1) {
153 if (p.deltab == 0) {
154 plaqpm(U, plaqp, d1, d2);
155 } else {
156 plaqpm_db(U, plaqp, d1, d2, p.deltab);
157 }
158
159 onsites(ALL) {
160 E[d1][X] -= eps * plaqp[X].project_to_algebra();
161 }
162 }
163 }

166template <typename T>
167void staplesum_db(const GaugeField<T> &U, Field<T> &staples, Direction d1, double deltab) {
168
169 Field<T> lower;
170
171 // zero staples just inc case
172 staples = 0;
173 foralldir(d2) if (d2 != d1) {
174
175 // anticipate that these are needed
176 // not really necessary, but may be faster
177 U[d2].start_gather(d1, ALL);
178 U[d1].start_gather(d2, ALL);
179
180 // calculate first lower 'U' of the staple sum
181 // do it on opp parity
182 onsites(ALL) {
183 double m;
184 if (2 * X.z() < lattice.size(e_z))
185 m = 1.0 - deltab;
186 else
187 m = 1.0 + deltab;
188
189 lower[X] = m * U[d2][X].dagger() * U[d1][X] * U[d2][X + d1];
190 }
191
192 // calculate then the upper 'n', and add the lower
193 // lower could also be added on a separate loop
194 onsites(ALL) {
195 double m;
196 if (2 * X.z() < lattice.size(e_z))
197 m = 1.0 - deltab;
198 else
199 m = 1.0 + deltab;
200
201 staples[X] += m * U[d2][X] * U[d1][X + d2] * U[d2][X + d1].dagger() + lower[X - d2];
202 }
203 }

206template <typename group>
207void update_E(const GaugeField<group> &U, VectorField<Algebra<group>> &E, const parameters &p,
208 double delta) {
209
210 Field<group> staple;
211 hila::scalar_type<group> eps = delta * p.beta / group::size();
212
213 foralldir(d) {
214 if (p.deltab != 0)
215 staplesum_db(U, staple, d, p.deltab);
216 else
217 staplesum(U, staple, d);
218
219 onsites(ALL) {
220 E[d][X] -= eps * (U[d][X] * staple[X].dagger()).project_to_algebra();
221 }
222 }
223}

241template <typename group>
242double measure_plaq_bp(const GaugeField<group> &U, double db = 0.0) {
243
244 Reduction<double> plaq;
245 plaq.allreduce(false).delayed(true);
246
247 foralldir(dir1) foralldir(dir2) if (dir1 < dir2) {
248 if (db == 0.0) {
249
250 onsites(ALL) {
251 plaq += real(trace(get_bp_iOsqmat(U[dir1][X] * U[dir2][X + dir1] *
252 (U[dir2][X] * U[dir1][X + dir2]).dagger()))) /
253 group::size();
254 }
255 } else {
256
257 onsites(ALL) {
258
259 double c;
260 if (2 * X.z() < lattice.size(e_z))
261 c = 1 - db;
262 else
263 c = 1 + db;
264
265 plaq += c *
266 real(trace(get_bp_iOsqmat(U[dir1][X] * U[dir2][X + dir1] *
267 (U[dir2][X] * U[dir1][X + dir2]).dagger()))) /
268 group::size();
269 }
270 }
271 }
272 return plaq.value();
273}

276template <typename group>
277double measure_plaq(const GaugeField<group> &U, double db = 0.0) {
278
279 Reduction<double> plaq;
280 plaq.allreduce(false);
281
282 foralldir(dir1) foralldir(dir2) if (dir1 < dir2) {
283 if (db == 0.0) {
284 onsites(ALL) {
285 plaq += 1.0 - real(trace(U[dir1][X] * U[dir2][X + dir1] *
286 U[dir1][X + dir2].dagger() * U[dir2][X].dagger())) /
287 group::size();
288 }
289 } else {
290 onsites(ALL) {
291 double c;
292 if (2 * X.z() < lattice.size(e_z))
293 c = 1 - db;
294 else
295 c = 1 + db;
296
297 plaq += c * (1.0 - real(trace(U[dir1][X] * U[dir2][X + dir1] *
298 U[dir1][X + dir2].dagger() * U[dir2][X].dagger())) /
299 group::size());
300 }
301 }
302 }
303 return plaq.value();
304}

306template <typename group>
307double measure_e2(const VectorField<Algebra<group>> &E) {
308
309 Reduction<double> e2 = 0;
310 e2.allreduce(false).delayed(true);
311
312 foralldir(d) {
313 onsites(ALL) e2 += E[d][X].squarenorm();
314 }
315 return e2.value() / 2;
316}

332template <typename T>
333void measure_polyakov_field(const GaugeField<T> &U, Field<float> &pl) {
334 Field<T> polyakov = U[e_t];
335
336 // mult links so that polyakov[X.dir == 0] contains the polyakov loop
337 for (int plane = lattice.size(e_t) - 2; plane >= 0; plane--) {
338
339 // safe_access(polyakov) pragma allows the expression below, otherwise
340 // hilapp would reject it because X and X+dir can refer to the same
341 // site on different "iterations" of the loop. However, here this
342 // is restricted on single dir-plane so it works but we must tell it to hilapp.
343
344#pragma hila safe_access(polyakov)
345 onsites(ALL) {
346 if (X.coordinate(e_t) == plane) {
347 polyakov[X] = U[e_t][X] * polyakov[X + e_t];
348 }
349 }
350 }
351
352 onsites(ALL) if (X.coordinate(e_t) == 0) {
353 pl[X] = real(trace(polyakov[X]));
354 }
355}

359void smear_polyakov_field(Field<float> &pl, int nsmear, float smear_coeff) {
360
361 if (nsmear > 0) {
362 Field<float> pl2 = 0;
363
364 for (int i = 0; i < nsmear; i++) {
365 onsites(ALL) if (X.coordinate(e_t) == 0) {
366 pl2[X] = pl[X] + smear_coeff * (pl[X + e_x] + pl[X - e_x] + pl[X + e_y] +
367 pl[X - e_y] + pl[X + e_z] + pl[X - e_z]);
368 }
369
370 onsites(ALL) if (X.coordinate(e_t) == 0) {
371 pl[X] = pl2[X] / (1 + 6 * smear_coeff);
372 }
373 }
374 }
375}

430template <typename group>
431void measure_polyakov_surface(const GaugeField<group> &U, const parameters &p, int traj) {
432
433 Field<float> pl;
434 measure_polyakov_field(U, pl);
435
436 std::vector<float> profile, profile_unsmear, prof1;
437 profile_unsmear = measure_polyakov_profile(pl, prof1);
438
439 smear_polyakov_field(pl, p.n_smear, p.smear_coeff);
440
441 if (p.z_smear > 0) {
442 Field<float> pl2 = 0;
443 for (int i = 0; i < p.z_smear; i++) {
444 onsites(ALL) if (X.coordinate(e_t) == 0) {
445 pl2[X] = pl[X] + p.smear_coeff * (pl[X + e_z] + pl[X - e_z]);
446 }
447 onsites(ALL) if (X.coordinate(e_t) == 0) {
448 pl[X] = pl2[X] / (1 + 2 * p.smear_coeff);
449 }
450 }
451 }
452
453 profile = measure_polyakov_profile(pl, prof1);
454
455 hila::out0 << std::setprecision(5);
456 double m = 1.0 / (lattice.size(e_x) * lattice.size(e_y));
457 for (int i = 0; i < profile.size(); i++) {
458 profile[i] *= m;
459 hila::out0 << "PRO " << i << ' ' << profile[i] << ' ' << prof1[i] << ' '
460 << profile_unsmear[i] * m << '\n';
461 }
462
463 float min = 1e8, max = 0;
464 int minloc, maxloc;
465 for (int i = 0; i < profile.size(); i++) {
466 if (min > profile[i]) {
467 min = profile[i];
468 minloc = i;
469 }
470 if (max < profile[i]) {
471 max = profile[i];
472 maxloc = i;
473 }
474 }
475
476 // find the surface between minloc and maxloc
477 float surface_level = max * 0.5; // assume min is really 0
478 int area = lattice.size(e_x) * lattice.size(e_y);
479
480 hila::out0 << "Surface level " << surface_level << '\n';
481
482 int startloc;
483 if (maxloc > minloc)
484 startloc = (maxloc + minloc) / 2;
485 else
486 startloc = ((maxloc + minloc + lattice.size(e_z)) / 2) % lattice.size(e_z);
487
488 std::vector<float> surf;
489 if (hila::myrank() == 0)
490 surf.resize(area);
491
492 hila::out0 << std::setprecision(6);
493
494 for (int y = 0; y < lattice.size(e_y); y++)
495 for (int x = 0; x < lattice.size(e_x); x++) {
496 auto line = pl.get_slice({x, y, -1, 0});
497 if (hila::myrank() == 0) {
498 // start search of the surface from the center between min and max
499 int z = startloc;
500 if (line[z] > surface_level) {
501 while (line[z_ind(z)] > surface_level && startloc - z < lattice.size(e_z) * 0.8)
502 z--;
503 } else {
504 while (line[z_ind(z + 1)] <= surface_level &&
505 z - startloc < lattice.size(e_z) * 0.8)
506 z++;
507 }
508
509
510 // do linear interpolation
511 surf[x + y * lattice.size(e_x)] =
512 z + (surface_level - line[z_ind(z)]) / (line[z_ind(z + 1)] - line[z_ind(z)]);
513
514 if (p.n_surface > 0 && (traj + 1) % p.n_surface == 0) {
515 hila::out0 << "SURF " << x << ' ' << y << ' ' << surf[x + y * lattice.size(e_x)]
516 << '\n';
517 }
518 }
519 }
520
521 if (hila::myrank() == 0) {
522 // do fft for the surface
523 static bool first = true;
524 static Complex<double> *buf;
525 static fftw_plan fftwplan;
526
527 if (first) {
528 first = false;
529
530 buf = (Complex<double> *)fftw_malloc(sizeof(Complex<double>) * area);
531
532 // note: we had x as the "fast" dimension, but fftw wants the 2nd dim to be
533 // the "fast" one. thus, first y, then x.
534 fftwplan = fftw_plan_dft_2d(lattice.size(e_y), lattice.size(e_x), (fftw_complex *)buf,
535 (fftw_complex *)buf, FFTW_FORWARD, FFTW_ESTIMATE);
536 }
537
538 for (int i = 0; i < area; i++) {
539 buf[i] = surf[i];
540 }
541
542 fftw_execute(fftwplan);
543
544 constexpr int pow_size = 200;
545
546 std::vector<double> npow(pow_size);
547 std::vector<int> hits(pow_size);
548
549 for (int i = 0; i < area; i++) {
550 int x = i % lattice.size(e_x);
551 int y = i / lattice.size(e_x);
552 x = (x <= lattice.size(e_x) / 2) ? x : (lattice.size(e_x) - x);
553 y = (y <= lattice.size(e_y) / 2) ? y : (lattice.size(e_y) - y);
554
555 int k = x * x + y * y;
556 if (k < pow_size) {
557 npow[k] += buf[i].squarenorm() / (area * area);
558 hits[k]++;
559 }
560 }
561
562 for (int i = 0; i < pow_size; i++) {
563 if (hits[i] > 0)
564 hila::out0 << "POW " << i << ' ' << npow[i] / hits[i] << ' ' << hits[i] << '\n';
565 }
566 }
567}

572template <typename group>
573void checkpoint(const GaugeField<group> &U, int trajectory, const parameters &p) {
574
575 double t = hila::gettime();
576 // save config
577 U.config_write(p.config_file);
578
579 // write run_status file
580 if (hila::myrank() == 0) {
581 std::ofstream outf;
582 outf.open("run_status", std::ios::out | std::ios::trunc);
583 outf << "trajectory " << trajectory + 1 << '\n';
584 outf << "seed " << static_cast<uint64_t>(hila::random() * (1UL << 61)) << '\n';
585 outf << "time " << hila::gettime() << '\n';
586 outf.close();
587 }
588
589 std::stringstream msg;
590 msg << "Checkpointing, time " << hila::gettime() - t;
591 hila::timestamp(msg.str().c_str());
592}

642template <typename group>
643void do_trajectory(GaugeField<group> &U, VectorField<Algebra<group>> &E, const parameters &p) {
644
645 update_U(U, E, p.dt / 2);
646 for (int n = 0; n < p.trajlen - 1; n++) {
647 update_E_bp(U, E, p, p.dt);
648 update_U(U, E, p.dt);
649 }
650 // and bring U and E to the same time value
651 update_E_bp(U, E, p, p.dt);
652 update_U(U, E, p.dt / 2);
653 regroup_gauge(U);
654}
