93 class Vec16f {
94 protected:
95  __m512 zmm; // Float vector
96 public:
97  // Default constructor:
98  Vec16f() = default;
99  // Constructor to broadcast the same value into all elements:
100  Vec16f(float f) {
101  zmm = _mm512_set1_ps(f);
102  }
103  // Constructor to build from all elements:
104  Vec16f(float f0, float f1, float f2, float f3, float f4, float f5, float f6, float f7,
105  float f8, float f9, float f10, float f11, float f12, float f13, float f14, float f15) {
106  zmm = _mm512_setr_ps(f0, f1, f2, f3, f4, f5, f6, f7, f8, f9, f10, f11, f12, f13, f14, f15);
107  }
108  // Constructor to build from two Vec8f:
109  Vec16f(Vec8f const a0, Vec8f const a1) {
110  zmm = _mm512_castpd_ps(_mm512_insertf64x4(_mm512_castps_pd(_mm512_castps256_ps512(a0)), _mm256_castps_pd(a1), 1));
111  }
112  // Constructor to convert from type __m512 used in intrinsics:
113  Vec16f(__m512 const x) {
114  zmm = x;
115  }
116  // Assignment operator to convert from type __m512 used in intrinsics:
117  Vec16f & operator = (__m512 const x) {
118  zmm = x;
119  return *this;
120  }
121  // Type cast operator to convert to __m512 used in intrinsics
122  operator __m512() const {
123  return zmm;
124  }
125  // Member function to load from array (unaligned)
126  Vec16f & load(float const * p) {
127  zmm = _mm512_loadu_ps(p);
128  return *this;
129  }
130  // Member function to load from array, aligned by 64
131  // You may use load_a instead of load if you are certain that p points to an address divisible by 64
132  Vec16f & load_a(float const * p) {
133  zmm = _mm512_load_ps(p);
134  return *this;
135  }
136  // Member function to store into array (unaligned)
137  void store(float * p) const {
138  _mm512_storeu_ps(p, zmm);
139  }
140  // Member function storing into array, aligned by 64
141  // You may use store_a instead of store if you are certain that p points to an address divisible by 64
142  void store_a(float * p) const {
143  _mm512_store_ps(p, zmm);
144  }
145  // Member function storing to aligned uncached memory (non-temporal store).
146  // This may be more efficient than store_a when storing large blocks of memory if it
147  // is unlikely that the data will stay in the cache until it is read again.
148  // Note: Will generate runtime error if p is not aligned by 16
149  void store_nt(float * p) const {
150  _mm512_stream_ps(p, zmm);
151  }
152  // Partial load. Load n elements and set the rest to 0
153  Vec16f & load_partial(int n, float const * p) {
154  zmm = _mm512_maskz_loadu_ps(__mmask16((1 << n) - 1), p);
155  return *this;
156  }
157  // Partial store. Store n elements
158  void store_partial(int n, float * p) const {
159  _mm512_mask_storeu_ps(p, __mmask16((1 << n) - 1), zmm);
160  }
161  // cut off vector to n elements. The last 8-n elements are set to zero
162  Vec16f & cutoff(int n) {
163  zmm = _mm512_maskz_mov_ps(__mmask16((1 << n) - 1), zmm);
164  return *this;
165  }
166  // Member function to change a single element in vector
167  Vec16f const insert(int index, float value) {
168  zmm = _mm512_mask_broadcastss_ps(zmm, __mmask16(1u << index), _mm_set_ss(value));
169  return *this;
170  }
171  // Member function extract a single element from vector
172  float extract(int index) const {
173  __m512 x = _mm512_maskz_compress_ps(__mmask16(1u << index), zmm);
174  return _mm512_cvtss_f32(x);
175  }
176  // Extract a single element. Use store function if extracting more than one element.
177  // Operator [] can only read an element, not write.
178  float operator [] (int index) const {
179  return extract(index);
180  }
181  // Member functions to split into two Vec4f:
182  Vec8f get_low() const {
183  return _mm512_castps512_ps256(zmm);
184  }
185  Vec8f get_high() const {
186  return _mm256_castpd_ps(_mm512_extractf64x4_pd(_mm512_castps_pd(zmm),1));
187  }
188  static constexpr int size() {
189  return 16;
190  }
191  static constexpr int elementtype() {
192  return 16;
193  }
194  typedef __m512 registertype;
195 };

534 // Each index i0 - i7 is 1 for changing sign on the corresponding element, 0 for no change

537  constexpr __mmask16 m = __mmask16((i0&1) | (i1&1)<<1 | (i2&1)<< 2 | (i3&1)<<3 | (i4&1)<<4 | (i5&1)<<5 | (i6&1)<<6 | (i7&1)<<7
538  | (i8&1)<<8 | (i9&1)<<9 | (i10&1)<<10 | (i11&1)<<11 | (i12&1)<<12 | (i13&1)<<13 | (i14&1)<<14 | (i15&1)<<15);
539  if constexpr ((uint16_t)m == 0) return a;
540  __m512 s = _mm512_castsi512_ps(_mm512_maskz_set1_epi32(m, 0x80000000));
541  return a ^ s;
542 }

1509 // Permute vector of 8 64-bit integers.
1510 template <int i0, int i1, int i2, int i3, int i4, int i5, int i6, int i7>
1511 static inline Vec8d permute8(Vec8d const a) {
1512  int constexpr indexs[8] = { i0, i1, i2, i3, i4, i5, i6, i7 }; // indexes as array
1513  __m512d y = a; // result
1514  // get flags for possibilities that fit the permutation pattern
1515  constexpr uint64_t flags = perm_flags<Vec8d>(indexs);
1516 
1517  static_assert((flags & perm_outofrange) == 0, "Index out of range in permute function");
1518 
1519  if constexpr ((flags & perm_allzero) != 0) return _mm512_setzero_pd(); // just return zero
1520 
1521  if constexpr ((flags & perm_perm) != 0) { // permutation needed
1522 
1523  if constexpr ((flags & perm_largeblock) != 0) { // use larger permutation
1524  constexpr EList<int, 4> L = largeblock_perm<8>(indexs); // permutation pattern
1525  constexpr uint8_t ppat = (L.a[0] & 3) | (L.a[1]<<2 & 0xC) | (L.a[2]<<4 & 0x30) | (L.a[3]<<6 & 0xC0);
1526  y = _mm512_shuffle_f64x2(a, a, ppat);
1527  }
1528  else if constexpr ((flags & perm_same_pattern) != 0) { // same pattern in all lanes
1529  if constexpr ((flags & perm_punpckh) != 0) { // fits punpckhi
1530  y = _mm512_unpackhi_pd(y, y);
1531  }
1532  else if constexpr ((flags & perm_punpckl)!=0){ // fits punpcklo
1533  y = _mm512_unpacklo_pd(y, y);
1534  }
1535  else { // general permute within lanes
1536  constexpr uint8_t mm0 = (i0&1) | (i1&1)<<1 | (i2&1)<<2 | (i3&1)<<3 | (i4&1)<<4 | (i5&1)<<5 | (i6&1)<<6 | (i7&1)<<7;
1537  y = _mm512_permute_pd(a, mm0); // select within same lane
1538  }
1539  }
1540  else { // different patterns in all lanes
1541  if constexpr ((flags & perm_rotate_big) != 0) { // fits big rotate
1542  constexpr uint8_t rot = uint8_t(flags >> perm_rot_count); // rotation count
1543  y = _mm512_castsi512_pd(_mm512_alignr_epi64 (_mm512_castpd_si512(y), _mm512_castpd_si512(y), rot));
1544  }
1545  else if constexpr ((flags & perm_broadcast) != 0) { // broadcast one element
1546  constexpr int e = flags >> perm_rot_count;
1547  if constexpr(e != 0) {
1548  y = _mm512_castsi512_pd(_mm512_alignr_epi64(_mm512_castpd_si512(y), _mm512_castpd_si512(y), e));
1549  }
1550  y = _mm512_broadcastsd_pd(_mm512_castpd512_pd128(y));
1551  }
1552  else if constexpr ((flags & perm_compress) != 0) {
1553  y = _mm512_maskz_compress_pd(__mmask8(compress_mask(indexs)), y); // compress
1554  if constexpr ((flags & perm_addz2) == 0) return y;
1555  }
1556  else if constexpr ((flags & perm_expand) != 0) {
1557  y = _mm512_maskz_expand_pd(__mmask8(expand_mask(indexs)), y); // expand
1558  if constexpr ((flags & perm_addz2) == 0) return y;
1559  }
1560  else if constexpr ((flags & perm_cross_lane) == 0) { // no lane crossing
1561  if constexpr ((flags & perm_zeroing) == 0) { // no zeroing. use vpermilps
1562  const __m512i pmask = constant16ui <i0<<1, 0, i1<<1, 0, i2<<1, 0, i3<<1, 0, i4<<1, 0, i5<<1, 0, i6<<1, 0, i7<<1, 0>();
1563  return _mm512_permutevar_pd(a, pmask);
1564  }
1565  else { // with zeroing. pshufb may be marginally better because it needs no extra zero mask
1566  constexpr EList <int8_t, 64> bm = pshufb_mask<Vec8q>(indexs);
1567  return _mm512_castsi512_pd(_mm512_shuffle_epi8(_mm512_castpd_si512(y), Vec8q().load(bm.a)));
1568  }
1569  }
1570  else {
1571  // full permute needed
1572  const __m512i pmask = constant16ui <
1573  i0 & 7, 0, i1 & 7, 0, i2 & 7, 0, i3 & 7, 0, i4 & 7, 0, i5 & 7, 0, i6 & 7, 0, i7 & 7, 0>();
1574  y = _mm512_permutexvar_pd(pmask, y);
1575  }
1576  }
1577  }
1578  if constexpr ((flags & perm_zeroing) != 0) { // additional zeroing needed
1579  y = _mm512_maskz_mov_pd(zero_mask<8>(indexs), y);
1580  }
1581  return y;
1582 }
