294 void allocate() {
295 assert(fs == nullptr);
296 if (lattice.volume() == 0) {
297 hila::out0 << "Can not allocate Field variables before lattice.setup()\n";
298 hila::terminate(0);
299 }
300 fs = (field_struct *)memalloc(sizeof(field_struct));
301 fs->lattice_id = lattice.id();
302 fs->allocate_payload();
303 fs->initialize_communication();
304 mark_changed(ALL); // guarantees communications will be done
305 fs->assigned_to = 0; // and this means that it is not assigned
306
307 for (Direction d = (Direction)0; d < NDIRS; ++d) {
308
309#if !defined(CUDA) && !defined(HIP)
310 fs->neighbours[d] = lattice.neighb[d];
311#else
312 fs->payload.neighbours[d] = lattice.backend_lattice->d_neighb[d];
313#endif
314 }
315
316#ifdef SPECIAL_BOUNDARY_CONDITIONS
317 foralldir(dir) {
318 fs->boundary_condition[dir] = hila::bc::PERIODIC;
319 fs->boundary_condition[-dir] = hila::bc::PERIODIC;
320 }
321#endif
322
323#ifdef VECTORIZED
324 if constexpr (hila::is_vectorizable_type<T>::value) {
325 fs->vector_lattice = lattice.backend_lattice
326 ->get_vectorized_lattice<hila::vector_info<T>::vector_size>();
327 } else {
328 fs->vector_lattice = nullptr;
329 }
330#endif
331 }

414 void mark_changed(const Parity p) const {
415
416 for (Direction i = (Direction)0; i < NDIRS; ++i) {
417 // check if there's ongoing comms, invalidate it!
418 drop_comms(i, opp_parity(p));
419
420 set_gather_status(opp_parity(p), i, gather_status_t::NOT_DONE);
421 if (p != ALL) {
422 set_gather_status(ALL, i, gather_status_t::NOT_DONE);
423 } else {
424 set_gather_status(EVEN, i, gather_status_t::NOT_DONE);
425 set_gather_status(ODD, i, gather_status_t::NOT_DONE);
426 }
427 }
428 fs->assigned_to |= parity_bits(p);
429 }

513 void set_boundary_condition(Direction dir, hila::bc bc) {
514
515#ifdef SPECIAL_BOUNDARY_CONDITIONS
516
517
518#ifdef HILAPP
519 // this section is just to generate loop function calls for unary-. Will removed by hilapp
520 // from final code. If type T does not have -, give error
521
522 static_assert(has_unary_minus<T>::value, "BC possible only for types which implement unary "
523 "minus (-) -operator and are not unsigned");
524
525 ensure_unary_minus_is_loop_function<T>();
526#endif
527
528 // TODO: This works as intended only for periodic/antiperiodic b.c.
529 check_alloc();
530 fs->boundary_condition[dir] = bc;
531 fs->boundary_condition[-dir] = bc;
532#if !defined(CUDA) && !defined(HIP)
533 fs->neighbours[dir] = lattice.get_neighbour_array(dir, bc);
534 fs->neighbours[-dir] = lattice.get_neighbour_array(-dir, bc);
535#else
536 if (bc == hila::bc::PERIODIC) {
537 fs->payload.neighbours[dir] = lattice.backend_lattice->d_neighb[dir];
538 fs->payload.neighbours[-dir] = lattice.backend_lattice->d_neighb[-dir];
539 } else {
540 fs->payload.neighbours[dir] = lattice.backend_lattice->d_neighb_special[dir];
541 fs->payload.neighbours[-dir] = lattice.backend_lattice->d_neighb_special[-dir];
542 }
543#endif
544
545 // Make sure boundaries get refreshed
546 mark_changed(ALL);
547#else
548 assert(bc == hila::bc::PERIODIC &&
549 "Only periodic bondary conditions when SPECIAL_BOUNDARY_CONDITIONS is undefined");
550#endif
551 }
