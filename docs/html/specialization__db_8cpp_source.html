31 void load_spec_db() {
32  std::ifstream db(specialization_db_filename);
33  if (db.is_open()) {
34  std::string line;
35  while (std::getline(db, line)) {
36  // skip commented lines (start with #)
37 
38  if (line.length() > 0 && remove_initial_whitespace(line)[0] != '#') {
39  spec s;
40  std::istringstream iss(line);
41  iss >> s.decl;
42  iss >> s.file;
43  iss >> s.timestamp;
44 
45  if (iss.fail()) {
46  llvm::errs() << program_name << ": I/O error in "
47  << specialization_db_filename << '\n';
48  if (iss.eof())
49  llvm::errs()
50  << "Too few arguments on a line \"" << line << "\"\n";
51  llvm::errs() << "Suggest removing " << specialization_db_filename
52  << " and recompile\n";
53  exit(1);
54  }
55  // do not include own specializations in database, will be added again
56 
57  if (s.file != global.main_file_name)
58  specializations.push_back(s);
59  }
60  }
61  db.close();
62  } else {
63  // db does not exist, nothing to do
64  specializations.clear();
65  }
66 }

76 // checks whether this decl exists in database. If so, returns true and filename in
77 // "here". if not, returns false and adds the item in db
78 bool in_specialization_db(const std::string &decl_in, std::string &here) {
79 
80  std::string decl = remove_extra_whitespace(decl_in);
81 
82  if (!db_loaded)
83  load_spec_db();
84  db_loaded = true;
85 
86  spec *s = search_spec_db(decl);
87  if (s == nullptr) {
88  // add new item
89  spec sp = {decl, global.main_file_name, std::time(nullptr)};
90  specializations.push_back(sp);
91  } else if (s->file != global.main_file_name) {
92  // somebody else has specialization already -- however, it is possible that
93  // it is waiting for compilation and it may not include it any more.
94 
95  struct stat status;
96 
97  if (stat(s->file.c_str(), &status) == 0 &&
98  #ifndef __APPLE__
99  status.st_mtim.tv_sec > s->timestamp
100  #else
101  status.st_mtimespec.tv_sec > s->timestamp
102  #endif
103  ) {
104  // If the mod time of the file is later than timestamp, future compilation
105  // expected: steal the specialization
106  s->timestamp = std::time(nullptr);
107  s->file = global.main_file_name;
108  } else {
109  // "found" exit branch
110  here = s->file;
111  return true;
112  }
113  } else {
114  // now main_file == s->file -- should not happen
115  llvm::errs() << program_name << ": specialization db handling error\n";
116  exit(1);
117  }
118 
119  here = global.main_file_name;
120  return false;
121 }

147  llvm::errs() << program_name << ": specialization db write error\n";

149  }
150

152  db_loaded = false;
