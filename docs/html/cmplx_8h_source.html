551// write also multiply-add directly with complex numbers
552template <typename T>
553inline Complex<T> mul_add(const Complex<T> &a, const Complex<T> &b, const Complex<T> &c) {
554 // a*b + c
555 Complex<T> r;
556 T t1 = mul_add(a.re, b.re, c.re);
557 T t2 = mul_add(a.re, b.im, c.im);
558 r.re = mul_add(a.im, b.im, t1); // -a.im*b.im + a.re*b.re + c.re
559 r.im = mul_add(a.im, b.re, t2); // a.im*b.re + a.re*b.im + c.im
560 return r;
561}

751template <typename A, typename T, std::enable_if_t<hila::contains_complex<T>::value, int> = 0>
752inline auto operator*(const Imaginary_t<A> &i, const T &c) {
753 Complex<hila::scalar_type<T>> ca, cb;
754 T res;
755 constexpr int n_cmplx = sizeof(T) / sizeof(Complex<hila::scalar_type<T>>);
756 for (int k = 0; k < n_cmplx; k++) {
757 ca = hila::get_complex_in_var(c, k);
758 cb.re = -ca.im * i.imag();
759 cb.im = ca.re * i.imag();
760 hila::set_complex_in_var(res, k, cb);
761 }
762 return res;
763}
