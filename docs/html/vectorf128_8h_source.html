46 // Select between two __m128 sources, element by element, with broad boolean vector.
47 // Corresponds to this pseudocode:
48 // for (int i = 0; i < 4; i++) result[i] = s[i] ? a[i] : b[i];
49 // Each element in s must be either 0 (false) or 0xFFFFFFFF (true).
50 // No other values are allowed for broad boolean vectors.
51 // The implementation depends on the instruction set:
52 // If SSE4.1 is supported then only bit 31 in each dword of s is checked,
53 // otherwise all bits in s are used.
54 static inline __m128 selectf(__m128 const s, __m128 const a, __m128 const b) {
55 #if INSTRSET >= 5 // SSE4.1 supported
56  return _mm_blendv_ps(b, a, s);
57 #else
58  return _mm_or_ps(
59  _mm_and_ps(s, a),
60  _mm_andnot_ps(s, b));
61 #endif
62 }
63 
64 // Same, with two __m128d sources.
65 // and operators. Corresponds to this pseudocode:
66 // for (int i = 0; i < 2; i++) result[i] = s[i] ? a[i] : b[i];
67 // Each element in s must be either 0 (false) or 0xFFFFFFFFFFFFFFFF (true). No other
68 // No other values are allowed for broad boolean vectors.
69 // The implementation depends on the instruction set:
70 // If SSE4.1 is supported then only bit 63 in each dword of s is checked,
71 // otherwise all bits in s are used.
72 static inline __m128d selectd(__m128d const s, __m128d const a, __m128d const b) {
73 #if INSTRSET >= 5 // SSE4.1 supported
74  return _mm_blendv_pd(b, a, s);
75 #else
76  return _mm_or_pd(
77  _mm_and_pd(s, a),
78  _mm_andnot_pd(s, b));
79 #endif
80 }

477 * Vec4f: Vector of 4 single precision floating point values

480 
481 class Vec4f {
482 protected:
483  __m128 xmm; // Float vector
484 public:
485  // Default constructor:
486  Vec4f() = default;
487  // Constructor to broadcast the same value into all elements:
488  Vec4f(float f) {
489  xmm = _mm_set1_ps(f);
490  }
491  // Constructor to build from all elements:
492  Vec4f(float f0, float f1, float f2, float f3) {
493  xmm = _mm_setr_ps(f0, f1, f2, f3);
494  }
495  // Constructor to convert from type __m128 used in intrinsics:
496  Vec4f(__m128 const x) {
497  xmm = x;
498  }
499  // Assignment operator to convert from type __m128 used in intrinsics:
500  Vec4f & operator = (__m128 const x) {
501  xmm = x;
502  return *this;
503  }
504  // Type cast operator to convert to __m128 used in intrinsics
505  operator __m128() const {
506  return xmm;
507  }
508  // Member function to load from array (unaligned)
509  Vec4f & load(float const * p) {
510  xmm = _mm_loadu_ps(p);
511  return *this;
512  }
513  // Member function to load from array, aligned by 16
514  // "load_a" is faster than "load" on older Intel processors (Pentium 4, Pentium M, Core 1,
515  // Merom, Wolfdale) and Atom, but not on other processors from Intel, AMD or VIA.
516  // You may use load_a instead of load if you are certain that p points to an address
517  // divisible by 16.
518  Vec4f & load_a(float const * p) {
519  xmm = _mm_load_ps(p);
520  return *this;
521  }
522  // Member function to store into array (unaligned)
523  void store(float * p) const {
524  _mm_storeu_ps(p, xmm);
525  }
526  // Member function storing into array, aligned by 16
527  // "store_a" is faster than "store" on older Intel processors (Pentium 4, Pentium M, Core 1,
528  // Merom, Wolfdale) and Atom, but not on other processors from Intel, AMD or VIA.
529  // You may use store_a instead of store if you are certain that p points to an address
530  // divisible by 16.
531  void store_a(float * p) const {
532  _mm_store_ps(p, xmm);
533  }
534  // Member function storing to aligned uncached memory (non-temporal store).
535  // This may be more efficient than store_a when storing large blocks of memory if it
536  // is unlikely that the data will stay in the cache until it is read again.
537  // Note: Will generate runtime error if p is not aligned by 16
538  void store_nt(float * p) const {
539  _mm_stream_ps(p, xmm);
540  }
541  // Partial load. Load n elements and set the rest to 0
542  Vec4f & load_partial(int n, float const * p) {
543 #if INSTRSET >= 10 // AVX512VL
544  xmm = _mm_maskz_loadu_ps(__mmask8((1u << n) - 1), p);
545 #else
546  switch (n) {
547  case 0:
548  *this = 0.f; break;
549  case 1:
550  xmm = _mm_load_ss(p); break;
551  case 2:
552  xmm = _mm_setr_ps(p[0], p[1], 0.f, 0.f); break;
553  case 3:
554  xmm = _mm_setr_ps(p[0], p[1], p[2], 0.f); break;
555  case 4:
556  load(p); break;
557  default:
558  break;
559  }
560 #endif
561  return *this;
562  }
563  // Partial store. Store n elements
564  void store_partial(int n, float * p) const {
565 #if INSTRSET >= 10 // AVX512VL
566  _mm_mask_storeu_ps(p, __mmask8((1u << n) - 1), xmm);
567 #else // storing in bigger blocks may unsafe unless compiler option -fno-strict-aliasing is specified,
568  // therefore we have to rely on the compiler to optimize this
569  float d[4];
570  store(d);
571  if (uint32_t(n) > 4) n = 4;
572  for (int i = 0; i < n; i++) {
573  p[i] = d[i];
574  }
575 #endif
576  }
577  // cut off vector to n elements. The last 4-n elements are set to zero
578  Vec4f & cutoff(int n) {
579 #if INSTRSET >= 10
580  xmm = _mm_maskz_mov_ps(__mmask8((1u << n) - 1), xmm);
581 #else
582  if (uint32_t(n) >= 4) return *this;
583  const union {
584  int32_t i[8];
585  float f[8];
586  } mask = { {-1,-1,-1,-1,0,0,0,0} };
587  xmm = _mm_and_ps(xmm, Vec4f().load(mask.f + 4 - n));
588 #endif
589  return *this;
590  }
591  // Member function to change a single element in vector
592  Vec4f const insert(int index, float value) {
593 #if INSTRSET >= 10 // AVX512VL
594  xmm = _mm_mask_broadcastss_ps(xmm, __mmask8(1u << index), _mm_set_ss(value));
595 #elif INSTRSET >= 5 // SSE4.1
596  switch (index & 3) {
597  case 0:
598  xmm = _mm_insert_ps(xmm, _mm_set_ss(value), 0 << 4); break;
599  case 1:
600  xmm = _mm_insert_ps(xmm, _mm_set_ss(value), 1 << 4); break;
601  case 2:
602  xmm = _mm_insert_ps(xmm, _mm_set_ss(value), 2 << 4); break;
603  default:
604  xmm = _mm_insert_ps(xmm, _mm_set_ss(value), 3 << 4); break;
605  }
606 #else
607  const int32_t maskl[8] = { 0,0,0,0,-1,0,0,0 };
608  __m128 broad = _mm_set1_ps(value); // broadcast value into all elements
609  __m128 mask = _mm_loadu_ps((float const*)(maskl + 4 - (index & 3))); // mask with FFFFFFFF at index position
610  xmm = selectf(mask, broad, xmm);
611 #endif
612  return *this;
613  }
614  // Member function extract a single element from vector
615  float extract(int index) const {
616 #if INSTRSET >= 10
617  __m128 x = _mm_maskz_compress_ps(__mmask8(1u << index), xmm);
618  return _mm_cvtss_f32(x);
619 #else
620  float x[4];
621  store(x);
622  return x[index & 3];
623 #endif
624  }
625  // Extract a single element. Use store function if extracting more than one element.
626  // Operator [] can only read an element, not write.
627  float operator [] (int index) const {
628  return extract(index);
629  }
630  static constexpr int size() {
631  return 4;
632  }
633  static constexpr int elementtype() {
634  return 16;
635  }
636  typedef __m128 registertype;
637 };

1087 template <typename VTYPE>
1088 static inline VTYPE pow_template_i(VTYPE const x0, int n) {
1089  VTYPE x = x0; // a^(2^i)
1090  VTYPE y(1.0f); // accumulator
1091  if (n >= 0) { // make sure n is not negative
1092  while (true) { // loop for each bit in n
1093  if (n & 1) y *= x; // multiply if bit = 1
1094  n >>= 1; // get next bit of n
1095  if (n == 0) return y; // finished
1096  x *= x; // x = a^2, a^4, a^8, etc.
1097  }
1098  }
1099  else {
1100  // n < 0
1101  if (uint32_t(n) == 0x80000000u) return nan_vec<VTYPE>(); // integer overflow
1102  return VTYPE(1.0f) / pow_template_i<VTYPE>(x0, -n); // reciprocal
1103  }

1123 
1124 // gcc can optimize pow_template_i to generate the same as the code below. MS and Clang can not.
1125 // Therefore, this code is kept

1127 template <typename V, int n>
1128 static inline V pow_n(V const a) {
1129  if (n == 0x80000000) return nan_vec<V>(); // integer overflow
1130  if (n < 0) return V(1.0f) / pow_n<V, -n>(a);
1131  if (n == 0) return V(1.0f);
1132  if (n >= 256) return pow(a, n);
1133  V x = a; // a^(2^i)
1134  V y; // accumulator
1135  const int lowest = n - (n & (n - 1)); // lowest set bit in n
1136  if (n & 1) y = x;
1137  if (n < 2) return y;
1138  x = x * x; // x^2
1139  if (n & 2) {
1140  if (lowest == 2) y = x; else y *= x;
1141  }
1142  if (n < 4) return y;
1143  x = x * x; // x^4
1144  if (n & 4) {
1145  if (lowest == 4) y = x; else y *= x;
1146  }
1147  if (n < 8) return y;
1148  x = x * x; // x^8
1149  if (n & 8) {
1150  if (lowest == 8) y = x; else y *= x;
1151  }
1152  if (n < 16) return y;
1153  x = x * x; // x^16
1154  if (n & 16) {
1155  if (lowest == 16) y = x; else y *= x;
1156  }
1157  if (n < 32) return y;
1158  x = x * x; // x^32
1159  if (n & 32) {
1160  if (lowest == 32) y = x; else y *= x;
1161  }
1162  if (n < 64) return y;
1163  x = x * x; // x^64

1166  }
1167  if (n < 128) return y;

2446  int constexpr indexs[2] = { i0, i1 }; // indexes as array
2447  __m128d y = a; // result
2448  // get flags for possibilities that fit the permutation pattern
2449  constexpr uint64_t flags = perm_flags<Vec2q>(indexs);
2450  static_assert((flags & perm_outofrange) == 0, "Index out of range in permute function");
2451  if constexpr ((flags & perm_allzero) != 0) return _mm_setzero_pd(); // just return zero
2452 
2453  constexpr bool fit_shleft = (flags & perm_shleft) != 0;
2454  constexpr bool fit_shright = (flags & perm_shright) != 0;
2455  constexpr bool fit_punpckh = (flags & perm_punpckh) != 0;
2456  constexpr bool fit_punpckl = (flags & perm_punpckl) != 0;
2457  constexpr bool fit_zeroing = (flags & perm_zeroing) != 0;
2458  if constexpr ((flags & perm_perm) != 0) { // permutation needed
2459  // try to fit various instructions
2460  if constexpr (fit_shleft && fit_zeroing) {
2461  // pslldq does both permutation and zeroing. if zeroing not needed use punpckl instead
2462  return _mm_castsi128_pd(_mm_bslli_si128(_mm_castpd_si128(a), 8));
2463  }
2464  if constexpr (fit_shright && fit_zeroing) {
2465  // psrldq does both permutation and zeroing. if zeroing not needed use punpckh instead
2466  return _mm_castsi128_pd(_mm_bsrli_si128(_mm_castpd_si128(a), 8));
2467  }
2468  if constexpr (fit_punpckh) { // fits punpckhi
2469  y = _mm_unpackhi_pd(a, a);
2470  }
2471  else if constexpr (fit_punpckl) { // fits punpcklo
2472  y = _mm_unpacklo_pd(a, a);
2473  }
2474  else { // needs general permute
2475  y = _mm_shuffle_pd(a, a, (i0 & 1) | (i1 & 1) * 2);
2476  }
2477  }
2478  if constexpr (fit_zeroing) {
2479  // additional zeroing needed
2480 #if INSTRSET >= 10 // use compact mask
2481  y = _mm_maskz_mov_pd(zero_mask<2>(indexs), y);
2482 #else // use unpack to avoid using data cache
2483  if constexpr (i0 == -1) {
2484  y = _mm_unpackhi_pd(_mm_setzero_pd(), y);
2485  }
2486  else if constexpr (i1 == -1) {
2487  y = _mm_unpacklo_pd(y, _mm_setzero_pd());
2488  }
2489 #endif
2490  }
2491  return y;
2492 }
