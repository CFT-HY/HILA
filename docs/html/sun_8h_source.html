128 void make_unitary() {
129
130 for (int r = 0; r < n; r++) {
131
132 // normalize row r
133 radix n2 = 0;
134 for (int c = 0; c < n; c++) n2 += this->e(r, c).squarenorm();
135 n2 = 1.0 / sqrt(n2);
136 for (int c = 0; c < n; c++) this->e(r, c) *= n2;
137
138 // Now make rows r+1 .. n-1 orthogonal to row r,
139 // by doing j = j - (r^* j) r
140
141 Complex<radix> d;
142 for (int j = r + 1; j < n; j++) {
143 // dot productof r^* j
144 d = 0;
145 for (int i = 0; i < n : i++) {
146 d += this->e(r, i).conj_mul(this->e(j, i));
147 }
148 // and j -= d * r
149 for (int i = 0; i < n : i++) {
150 this->e(j, i) -= d * this->e(r, i);
151 }
152 }
153 }

194 void random(const int depth = 12) {
195 Matrix<n, n, Complex<radix>> A, An, res;
196 An = 1;
197 res = 1;
198 Complex<radix> tr(1, 0), factor(1, 0);
199 for (int i = 0; i < n; i++) {
200 A.e(i, i) = Complex<radix>(hila::random(), 0.0);
201 for (int j = 0; j < i; j++) {
202 Complex<radix> a(static_cast<radix>(hila::random() / n),
203 static_cast<radix>(hila::random() / n));
204 A.e(i, j) = a;
205 A.e(j, i) = a.conj();
206 }
207 }
208 tr = A.trace() * (static_cast<radix>(1) / static_cast<radix>(n));
209 for (int i = 0; i < n; i++) {
210 A.e(i, i) -= tr;
211 }
212 An = A;
213 for (int k = 1; k <= depth; k++) {
214 factor = factor * Complex<radix>(0, 1) *
215 (static_cast<radix>(1) / static_cast<radix>(k));
216 res += An * factor;
217 An *= A;
218 }
219 for (int i = 0; i < n; i++)
220 for (int j = 0; j < n; j++) {
221 (*this).e(i, j) = res.e(i, j);

267 Complex<radix> det_lu() {
268
269 int i, imax, j, k;
270 radix big, d, temp, dum;
271 Complex<radix> cdum, csum, ctmp1;
272 radix vv[n];
273 Complex<radix> a[n][n];
274 Complex<radix> one;
275
276 one = Complex<radix>(1, 0);
277
278 d = 1;
279
280 imax = -1;
281
282 for (i = 0; i < n; i++)
283 for (j = 0; j < n; j++) a[i][j] = this->e(i, j);
284
285 for (i = 0; i < n; i++) {
286 big = 0;
287 for (j = 0; j < n; j++) {
288 if ((temp = a[i][j].squarenorm()) > big) big = temp;
289 }
290 big = sqrt(big);
291 if (big == 0.0) return 0;
292 vv[i] = 1.0 / big;
293 }
294
295 for (j = 0; j < n; j++) {
296 for (i = 0; i < j; i++) {
297 csum = a[i][j];
298 for (k = 0; k < i; k++) {
299 csum -= a[i][k] * a[k][j];
300 }
301 a[i][j] = csum;
302 }
303
304 big = 0;
305 for (i = j; i < n; i++) {
306 csum = a[i][j];
307 for (k = 0; k < j; k++) {
308 csum -= a[i][k] * a[k][j];
309 }
310 a[i][j] = csum;
311 if ((dum = vv[i] * csum.abs()) >= big) {
312 big = dum;
313 imax = i;
314 }
315 }
316
317 if (j != imax) {
318 for (k = 0; k < n; k++) {
319 cdum = a[imax][k];
320 a[imax][k] = a[j][k];
321 a[j][k] = cdum;
322 }
323 d = -d;
324 vv[imax] = vv[j];
325 }
326
327 if (a[j][j].abs() == static_cast<radix>(0.0))
328 a[j][j] = Complex<radix>(1e-20, 0);
329
330 if (j != n - 1) {
331 cdum = one / a[j][j]; // check Complex division
332 for (i = j + 1; i < n; i++) {
333 a[i][j] = a[i][j] * cdum;
334 }
335 }
336 }
337
338 csum = d;
339 for (j = 0; j < n; j++) {
340 csum = csum * a[j][j];
341 }
342
343 return (csum);
344 }

346 static SU generator(int ng) {
347 // SUN generators normalized as tr(T^2) = 1/2
348 static bool initialize = true;
349 static SU generators[n * n - 1];
350 if (initialize)
351 for (int g = 0; g < n * n - 1; g++) {
352 generators[g] = 0;
353 if (g < n - 1) {
354 // Diagonal generators
355 double w = 0.5 / sqrt((g + 1) * (g + 2) * 0.5);
356 for (int i = 0; i < g + 1; i++) {
357 generators[g].e(i, i).re = w;
358 }
359 generators[g].e(g + 1, g + 1).re = -(g + 1) * w;
360 } else {
361 // Nondiagonal ones. Just run through the indexes and
362 // count until they match...
363 int k = n - 1;
364 for (int m1 = 0; m1 < n; m1++)
365 for (int m2 = m1 + 1; m2 < n; m2++) {
366 if (g == k) {
367 generators[g].e(m1, m2).re = 0.5;
368 generators[g].e(m2, m1).re = 0.5;
369 } else if (g == k + 1) {
370 generators[g].e(m1, m2).im = 0.5;
371 generators[g].e(m2, m1).im = -0.5;
372 }
373 k += 2;
374 }
375 }
376 initialize = false;
377 }
378 return generators[ng];
379 }

382 return n * n - 1;

646template <int N, typename radix>
647void project_antihermitean(Matrix<N, N, Complex<radix>> &matrix) {
648 radix tr = 0;
649 for (int i = 0; i < N; i++) {
650 for (int j = 0; j < i; j++) {
651 radix a = 0.5 * (matrix.e(i, j).re - matrix.e(j, i).re);
652 radix b = 0.5 * (matrix.e(i, j).im + matrix.e(j, i).im);
653 matrix.e(i, j).re = a;
654 matrix.e(j, i).re = -a;
655 matrix.e(i, j).im = b;
656 matrix.e(j, i).im = b;
657 }
658 tr += matrix.e(i, i).im;
659 matrix.e(i, i).re = 0;
660 }
661 for (int i = 0; i < N; i++) {
662 matrix.e(i, i).im -= tr / N;
663 }
