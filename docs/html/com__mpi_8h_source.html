65template <typename T>
66MPI_Datatype get_MPI_number_type(size_t &size, bool with_int = false) {
67
68 if (std::is_same<hila::scalar_type<T>, int>::value) {
69 size = sizeof(int);
70 return with_int ? MPI_2INT : MPI_INT;
71 } else if (std::is_same<hila::scalar_type<T>, unsigned>::value) {
72 size = sizeof(unsigned);
73 return with_int ? MPI_2INT : MPI_UNSIGNED; // MPI does not contain MPI_UNSIGNED_INT
74 } else if (std::is_same<hila::scalar_type<T>, long>::value) {
75 size = sizeof(long);
76 return with_int ? MPI_LONG_INT : MPI_LONG;
77 } else if (std::is_same<hila::scalar_type<T>, int64_t>::value) {
78 size = sizeof(int64_t);
79 return with_int ? MPI_LONG_INT : MPI_INT64_T; // need to use LONG_INT
80 } else if (std::is_same<hila::scalar_type<T>, uint64_t>::value) {
81 size = sizeof(uint64_t);
82 return with_int ? MPI_LONG_INT : MPI_UINT64_T; // ditto
83 } else if (std::is_same<hila::scalar_type<T>, float>::value) {
84 size = sizeof(float);
85 return with_int ? MPI_FLOAT_INT : MPI_FLOAT;
86 } else if (std::is_same<hila::scalar_type<T>, double>::value) {
87 size = sizeof(double);
88 return with_int ? MPI_DOUBLE_INT : MPI_DOUBLE;
89 } else if (std::is_same<hila::scalar_type<T>, long double>::value) {
90 size = sizeof(long double);
91 return with_int ? MPI_LONG_DOUBLE_INT : MPI_LONG_DOUBLE;
92 }
93
94 size = 1;
95 return MPI_BYTE;
96}

137T broadcast(T &var, int rank = 0) {
138 static_assert(std::is_trivial<T>::value, "broadcast(var) must use trivial type");
139 if (hila::check_input)
140 return var;
141
142 assert(0 <= rank && rank < hila::number_of_nodes() && "Invalid sender rank in broadcast()");
143
144 broadcast_timer.start();
145 MPI_Bcast(&var, sizeof(T), MPI_BYTE, 0, lattice.mpi_comm_lat);
146 broadcast_timer.stop();
147 return var;

158template <typename T>
159void broadcast(std::vector<T> &list, int rank = 0) {
160
161 static_assert(std::is_trivial<T>::value, "broadcast(vector<T>) must have trivial T");
162
163 if (hila::check_input)
164 return;
165
166 broadcast_timer.start();
167
168 int size = list.size();
169 MPI_Bcast(&size, sizeof(int), MPI_BYTE, rank, lattice.mpi_comm_lat);
170 if (hila::myrank() != rank) {
171 list.resize(size);
172 }
173
174 // move vectors directly to the storage
175 MPI_Bcast((void *)list.data(), sizeof(T) * size, MPI_BYTE, rank, lattice.mpi_comm_lat);
176
177 broadcast_timer.stop();
178}

280template <typename T>
281void reduce_node_sum(T *value, int send_count, bool allreduce = true) {
282 T recv_data[send_count];
283 MPI_Datatype dtype;
284
285 if (hila::check_input)
286 return;
287
288 dtype = get_MPI_number_type<T>();
289
290 reduction_timer.start();
291 if (allreduce) {
292 MPI_Allreduce((void *)value, (void *)recv_data,
293 send_count * sizeof(T) / sizeof(hila::scalar_type<T>), dtype, MPI_SUM,
294 lattice.mpi_comm_lat);
295 for (int i = 0; i < send_count; i++)
296 value[i] = recv_data[i];
297 } else {
298 MPI_Reduce((void *)value, (void *)recv_data,
299 send_count * sizeof(T) / sizeof(hila::scalar_type<T>), dtype, MPI_SUM, 0,
300 lattice.mpi_comm_lat);
301 if (hila::myrank() == 0)
302 for (int i = 0; i < send_count; i++)
303 value[i] = recv_data[i];
304 }
305 reduction_timer.stop();
306}
