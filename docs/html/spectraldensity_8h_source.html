24inline int sd_get_k_bin(const CoordinateVector &cv, const sd_k_bin_parameters &p) {

248 void sd_calculate_bin_info() {
249
250 k_avg.resize(par.bins);
251 bin_count.resize(par.bins);
252
253 ReductionVector<double> s(par.bins);
254 ReductionVector<long> count(par.bins);
255 s = 0;
256 count = 0;
257
258 onsites(ALL) {
259
260 double kr = convert_to_k(X.coordinates()).norm();
261 int b = sd_get_k_bin(X.coordinates(), par);
262
263 if (b >= 0 && b < par.bins) {
264 s[b] += kr;
265 count[b] += 1;
266 }
267 }
268
269 for (int i = 0; i < par.bins; i++) {
270 bin_count[i] = count[i];
271 k_avg[i] = s[i] / count[i];
272 }
273 }
274
276
277 double k(int i) {
278
279 if (k_avg.size() == 0) {
280 sd_calculate_bin_info();
281 }
282
283 if (i >= 0 && i < par.bins)
284 return k_avg[i];
285 else
286 return 0.0;
287 }
288
290
291 long count(int i) {
292
293 if (bin_count.size() == 0) {
294 sd_calculate_bin_info();
295 }
296
297 if (i >= 0 && i < par.bins)
298 return bin_count[i];
299 else
300 return 0;
301 }
