48 bool LoopFunctionHandler::VisitVarDecl(VarDecl *var) {
49  std::string typestring = var->getType().getAsString();
50 
51  size_t begin;
52  begin = typestring.rfind("element", 0);
53  if (begin != std::string::npos) {
54  // This variable is an element, replace with vector
55  std::string vector_type = typestring;
56  if (begin != std::string::npos) {
57  vector_type.replace(begin, 7, "vectorize_struct");
58  vector_type.replace(vector_type.find_last_of(">"), 1,
59  ", " + std::to_string(vector_size) + ">::type");
60  }
61 
62  if (var->isDirectInit()) {
63  std::string init = TheRewriter.getRewrittenText(var->getInit()->getSourceRange());
64  functionBuffer.replace(var->getSourceRange(),
65  vector_type + " " + var->getNameAsString() + "=" + init);
66  } else {
67  functionBuffer.replace(var->getSourceRange(),
68  vector_type + " " + var->getNameAsString());
69  }
70 
71  } else {
72  if (var->hasInit()) {
73  LoopAssignChecker lac(*this);
74  lac.TraverseStmt(var->getInit());
75  }
76  }
77  return true;
78 }

124 void GeneralVisitor::handle_loop_function_avx(call_info_struct &ci) {
125 
126  FunctionDecl *fd = ci.funcdecl;
127 
128  SourceRange sr = fd->getSourceRange();
129  srcBuf *sourceBuf = get_file_srcBuf(sr.getBegin());
130 
131  if (sourceBuf == nullptr) {
132  // it's a system function - should probably do something?
133  return;
134  }
135 
136  // Track wether the function actually contains elements.
137  // if not, no new function should be written
138  bool generate_function = false;
139 
140  // Check allowed vector sizes
141  int smallest = 1, largest = 0;
142  for (clang::ParmVarDecl *par : fd->parameters()) {
143  std::string typestring = par->getType().getAsString(PP);
144  if (find_word(typestring, "double") != std::string::npos) {
145  smallest = 4;
146  largest = 8;
147  }
148  if (find_word(typestring, "float") != std::string::npos ||
149  find_word(typestring, "int") != std::string::npos ||
150  find_word(typestring, "CoordinateVector") != std::string::npos) {
151  smallest = 8;
152  largest = 16;
153  }
154 
155  // Check if there are elements in the first place
156  // THIS NEVER HAPPENS NOW; TODO: PROPER LOOP FUNCTION "rewrite"
157  if (find_word(typestring, "element") != std::string::npos)
158  generate_function = true;
159  }
160 
161  if (generate_function)
162  for (int vector_size = smallest; vector_size <= largest; vector_size *= 2) {
163 
164  LoopFunctionHandler lfh(*this);
165  lfh.functionBuffer.copy_from_range(sourceBuf, sr);
166  lfh.vector_size = vector_size;
167 
168  // Handle each parameter
169  for (clang::ParmVarDecl *par : fd->parameters()) {
170  std::string typestring = par->getType().getAsString(PP);
171  replace_element_with_vector(par->getSourceRange(), typestring,
172  par->getNameAsString(), vector_size,
173  lfh.functionBuffer);
174  }
175 
176  // Handle return type
177  // Note: C++ cannot specialize only based on return type. Therefore we
178  // only write a new function if the parameters contain elements
179  std::string typestring = fd->getReturnType().getAsString(PP);
180  replace_element_with_vector(fd->getReturnTypeSourceRange(), typestring, "", vector_size,
181  lfh.functionBuffer);
182 
183  lfh.TraverseStmt(fd->getBody());
184 
185  std::string buffer = lfh.functionBuffer.dump();
186  if (!(fd->hasBody())) {
187  // Declaration does not contain a body, needs a semicolon
188  buffer += ";";
189  }
190  buffer += "\n";
191  sourceBuf->insert(sr.getBegin(), buffer, true, true);
192  }
193 }

211 bool TopLevelVisitor::check_loop_vectorizable(Stmt *S, int &vector_size, std::string &diag_str) {
212 
213  vector_size = 0;
214  number_type numtype;
215  bool is_vectorizable = true;
216 
217  std::vector<std::string> reason = {};
218 
219  // check if loop has conditional
220  if (loop_info.has_pragma_novector) {
221  is_vectorizable = false;
222  reason.push_back("it has '#pragma hila novector'");
223 
224  } else {
225 
226  if (loop_info.has_site_dependent_cond_or_index) {
227  is_vectorizable = false;
228  reason.push_back("it contains site dependent conditional or array index");
229  }
230 
231  if (contains_random(S)) {
232  is_vectorizable = false;
233  reason.push_back("it contains a random number generator");
234  }
235 
236  if (selection_info_list.size() > 0) {
237  is_vectorizable = false;
238  reason.push_back("it contains site selection variable");
239  }
240 
241  std::string vector_var_name; // variable which determines the vectorization
242  std::string vector_var_type; // type of variable which determines the vectorization
243 
244  // check if the fields are vectorizable in a compatible manner
245  if (field_info_list.size() > 0) {
246  for (field_info &fi : field_info_list) {
247  if (!fi.vecinfo.is_vectorizable) {
248  is_vectorizable = false;
249  reason.push_back("Field variable '" + fi.old_name + "' is not vectorizable");
250  } else {
251  if (vector_size == 0) {
252  vector_size = fi.vecinfo.vector_size;
253  numtype = fi.vecinfo.basetype;
254  vector_var_name = fi.old_name;
255  vector_var_type = fi.vecinfo.basetype_str;
256  } else if (fi.vecinfo.vector_size != vector_size ||
257  fi.vecinfo.basetype != numtype) {
258  // TODO: let different vector types coexist!
259 
260  is_vectorizable = false;
261 
262  reason.push_back("type of variable '" + fi.old_name + "' is " +
263  fi.vecinfo.basetype_str + " and '" + vector_var_name +
264  "' is " + vector_var_type);
265  }
266  }
267  }
268  }
269 
270  // and then if the site dep. variables are vectorizable
271  if (var_info_list.size() > 0) {
272  for (var_info &vi : var_info_list)
273  if (!vi.is_raw && vi.is_site_dependent) {
274  if (vi.vecinfo.is_vectorizable) {
275  if (vector_size == 0) {
276  vector_size = vi.vecinfo.vector_size;
277  numtype = vi.vecinfo.basetype;
278  vector_var_name = vi.name;
279  vector_var_type = vi.vecinfo.basetype_str;
280 
281  } else if (vector_size != vi.vecinfo.vector_size ||
282  numtype != vi.vecinfo.basetype) {
283  is_vectorizable = false;
284 
285  reason.push_back("type of variables '" + vi.name + "' is " +
286  vi.vecinfo.basetype_str + " and '" + vector_var_name +
287  "' is " + vector_var_type);
288  }
289  } else {
290  is_vectorizable = false;
291 
292  reason.push_back("variable '" + vi.name + "' is not vectorizable");
293  }
294  }
295  }
296 
297  // and still, check the special functions
298  if (is_vectorizable) {
299  for (auto const &sfc : special_function_call_list) {
300  if (sfc.name == "coordinates" || sfc.name == "coordinate") {
301  is_vectorizable = false;
302 
303  reason.push_back(
304  "X.coordinates() and X.coordinate() make expression site dependent");
305 
306  // // returning int vector
307  // if (vector_size == 0) {
308  // vector_size = target.vector_size / sizeof(int);
309  // } else if (vector_size != target.vector_size / sizeof(int) ||
310  // numtype != number_type::INT) {
311  // is_vectorizable = false;
312 
313  // reason.push_back("functions 'X.coordinates()' and "
314  // "'X.coordinate(Direction)' return int, "
315  // "which is not compatible with " +
316  // vector_var_type + " vectors");
317  // }
318 
319  } else if (sfc.name == "parity") {
320  is_vectorizable = false;
321 
322  reason.push_back("function 'X.parity()' is not AVX vectorizable");
323 
324  } else if (sfc.name == "random" || sfc.name == "hila::random") {
325  is_vectorizable = false;
326  reason.push_back("random number generators prevent vectorization");
327  }
328  }
329  }
330 
331  // and check function calls
332  for (auto &ci : loop_function_calls) {
333 
337 
338  // This below works but prevents now vectorization in almost everywhere
339  // if (ci.is_site_dependent || !ci.is_vectorizable) {
340  if (ci.is_site_dependent && !ci.is_vectorizable) {
341  is_vectorizable = false;
342 
343  if (ci.funcdecl != nullptr) {
344  reason.push_back("loop contains function " + ci.funcdecl->getNameAsString() +
345  " which is not vectorizable");
346  } else if (ci.ctordecl != nullptr) {
347  reason.push_back("loop contains constructor " + ci.ctordecl->getNameAsString() +
348  " which is not vectorizable");
349  }
350  }
351  }
352  }
353 
354  if (vector_size == 0 && is_vectorizable) {
355  // super-special case - loop does not contain fields or anything site dependent.
356  // Loop is probably useless. Let us just not vectorize it.
357  is_vectorizable = false;
358  reason.push_back("loop does not seem to have site dependent content");
359  }
360 
361  if (!is_vectorizable) {
362  diag_str = "loop is not AVX vectorizable because:";
363  for (auto &s : reason)
364  diag_str += "\n " + s;
365 
366  if (cmdline::avx_info > 0 || cmdline::verbosity > 0)
367  reportDiag(DiagnosticsEngine::Level::Remark, S->getSourceRange().getBegin(), "%0",
368  diag_str.c_str());
369 
370  } else {
371  diag_str = "loop is AVX vectorizable";
372 
373  if (cmdline::avx_info > 1 || cmdline::verbosity > 1)
374  reportDiag(DiagnosticsEngine::Level::Remark, S->getSourceRange().getBegin(), "%0",
375  diag_str.c_str());
376  }
377 
378  return is_vectorizable;
379 }
