15template <typename T>
16void Field<T>::field_struct::gather_comm_elements(
17 Direction d, Parity par, T *RESTRICT buffer,
18 const lattice_struct::comm_node_struct &to_node) const {
19#ifndef VECTORIZED
20#ifdef SPECIAL_BOUNDARY_CONDITIONS
21 // note: -d in is_on_edge, because we're about to send stuff to that
22 // Direction (gathering from Direction +d)
23 if (boundary_condition[d] == hila::bc::ANTIPERIODIC && lattice.mynode.is_on_edge(-d)) {
24 payload.gather_comm_elements(buffer, to_node, par, lattice, true);
25 } else {
26 payload.gather_comm_elements(buffer, to_node, par, lattice, false);
27 }
28#else
29 payload.gather_comm_elements(buffer, to_node, par, lattice, false);
30#endif
31
32#else
33 // this is vectorized branch
34 bool antiperiodic = false;
35#ifdef SPECIAL_BOUNDARY_CONDITIONS
36 if (boundary_condition[d] == hila::bc::ANTIPERIODIC && lattice.mynode.is_on_edge(-d)) {
37 antiperiodic = true;
38 }
39#endif
40
41 if constexpr (hila::is_vectorizable_type<T>::value) {
42 // now vectorized layout
43 if (vector_lattice->is_boundary_permutation[abs(d)]) {
44 // with boundary permutation need to gather elems 1-by-1
45 int n;
46 const unsigned *index_list = to_node.get_sitelist(par, n);
47 if (!antiperiodic) {
48 payload.gather_elements(buffer, index_list, n, lattice);
49 } else {
50 payload.gather_elements_negated(buffer, index_list, n, lattice);
51 }
52 } else {
53 // without it, can do the full block
54 payload.gather_comm_vectors(buffer, to_node, par, vector_lattice, antiperiodic);
55 }
56 } else {
57 // not vectoizable, standard methods
58 int n;
59 const unsigned *index_list = to_node.get_sitelist(par, n);
60 if (!antiperiodic)
61 payload.gather_elements(buffer, index_list, n, lattice);
62 else {
63 payload.gather_elements_negated(buffer, index_list, n, lattice);
64 }
65 }
66#endif
67}

114template <typename T>
115T *Field<T>::field_struct::get_receive_buffer(Direction d, Parity par,
116 const lattice_struct::comm_node_struct &from_node) {
117#if defined(VANILLA)
118
119 return (T *)payload.get_buffer() + from_node.offset(par);
120
121#elif defined(CUDA) || defined(HIP)
122
123 unsigned offs = 0;
124 if (par == ODD)
125 offs = from_node.sites / 2;
126 if (receive_buffer[d] == nullptr) {
127 receive_buffer[d] = payload.allocate_mpi_buffer(from_node.sites);
128 }
129 return receive_buffer[d] + offs;
130
131#elif defined(VECTORIZED)
132
133 if constexpr (!hila::is_vectorizable_type<T>::value) {
134 // use vanilla type, field laid out in std fashion
135 return (T *)payload.get_buffer() + from_node.offset(par);
136 } else {
137 unsigned offs = 0;
138 if (par == ODD)
139 offs = from_node.sites / 2;
140
141 if (vector_lattice->is_boundary_permutation[abs(d)]) {
142 // extra copy operation needed
143 if (receive_buffer[d] == nullptr) {
144 receive_buffer[d] = payload.allocate_mpi_buffer(from_node.sites);
145 }
146 return receive_buffer[d] + offs;
147 } else {
148 // directly to halo buffer
149 constexpr unsigned vector_size = hila::vector_info<T>::vector_size;
150 return ((T *)payload.get_buffer() +
151 (vector_lattice->halo_offset[d] * vector_size + offs));
152 }
153 }
154#endif
155} // end of get_receive_buffer

161template <typename T>
162Field<T> &Field<T>::shift(const CoordinateVector &v, Field<T> &res, const Parity par) const {
163
164 // use this to store remaining moves
165 CoordinateVector rem = v;
166
167 // check the parity of the move
168 Parity par_s;
169
170 int len = 0;
171 foralldir(d) len += abs(rem[d]);
172
173 // no move, just copy field
174 if (len == 0) {
175 res = *this;
176 return res;
177 }
178
179 // opp_parity(ALL) == ALL
180 if (len % 2 == 0)
181 par_s = opp_parity(par);
182 else
183 par_s = par;
184
185 // is this already gathered from one of the dirs in v?
186 bool found_dir = false;
187 Direction mdir;
188 foralldir(d) {
189 if (rem[d] > 0 && gather_status(par_s, d) != gather_status_t::NOT_DONE) {
190 mdir = d;
191 found_dir = true;
192 break;
193 } else if (rem[d] < 0 && gather_status(par_s, -d) != gather_status_t::NOT_DONE) {
194 mdir = -d;
195 found_dir = true;
196 break;
197 }
198 }
199
200 if (!found_dir) {
201 // now did not find a 'ready' dir. Take the 1st available
202 foralldir(d) {
203 if (rem[d] > 0) {
204 mdir = d;
205 break;
206 } else if (rem[d] < 0) {
207 mdir = -d;
208 break;
209 }
210 }
211 }
212
213 // Len 1, copy directly
214 if (len == 1) {
215 res[par_s] = (*this)[X + mdir];
216 return res;
217 }
218
219 // now longer - need buffer
220 Field<T> r1;
221 Field<T> *from, *to;
222
223 // this ensures that the final move lands on res
224 if (len % 2 == 0) {
225 from = &r1;
226 to = &res;
227 } else {
228 from = &res;
229 to = &r1;
230 }
231 // and copy initially to "from"
232 (*from)[par_s] = (*this)[X + mdir];
233
234 // and subtract remaining moves from rem
235 rem = rem - mdir;
236 par_s = opp_parity(par_s);
237
238 foralldir(d) {
239 if (rem[d] != 0) {
240 mdir = (rem[d] > 0) ? d : -d;
241
242 while (rem[d] != 0) {
243
244 (*to)[par_s] = (*from)[X + mdir];
245
246 par_s = opp_parity(par_s);
247 rem = rem - mdir;
248 std::swap(to, from);
249 }
250 }
251 }
252
253 return res;
254}

261template <typename T>
262dir_mask_t Field<T>::start_gather(Direction d, Parity p) const {
263
264 // get the mpi message tag right away, to ensure that we are always synchronized
265 // with the mpi calls -- some nodes might not need comms, but the tags must be in
266 // sync
267
268 int tag = get_next_msg_tag();
269
270 lattice_struct::nn_comminfo_struct &ci = lattice.nn_comminfo[d];
271 lattice_struct::comm_node_struct &from_node = ci.from_node;
272 lattice_struct::comm_node_struct &to_node = ci.to_node;
273
274 // check if this is done - either gathered or no comm to be done in the 1st place
275
276 if (is_gathered(d, p)) {
277 lattice.n_gather_avoided++;
278 return 0; // nothing to wait for
279 }
280
281 // No comms to do, nothing to wait for -- we'll use the is_gathered
282 // status to keep track of vector boundary shuffle anyway
283
284 if (from_node.rank == hila::myrank() && to_node.rank == hila::myrank()) {
285 fs->set_local_boundary_elements(d, p);
286 mark_gathered(d, p);
287 return 0;
288 }
289
290 // if this parity or ALL-type gather is going on nothing to be done
291 if (!gather_not_done(d, p) || !gather_not_done(d, ALL)) {
292 lattice.n_gather_avoided++;
293 return get_dir_mask(d); // nothing to do, but still need to wait
294 }
295
296 Parity par = p;
297 // if p is ALL but ODD or EVEN is going on/done, turn off parity which is not needed
298 // corresponding wait must do the same thing
299 if (p == ALL) {
300 if (!gather_not_done(d, EVEN) && !gather_not_done(d, ODD)) {
301 // even and odd are going on or ready, nothing to be done
302 lattice.n_gather_avoided++;
303 return get_dir_mask(d);
304 }
305 if (!gather_not_done(d, EVEN))
306 par = ODD;
307 else if (!gather_not_done(d, ODD))
308 par = EVEN;
309 // if neither is the case par = ALL
310 }
311
312 mark_gather_started(d, par);
313
314 // Communication hasn't been started yet, do it now
315
316 int par_i = static_cast<int>(par) - 1; // index to dim-3 arrays
317
318 constexpr size_t size = sizeof(T);
319
320 T *receive_buffer;
321 T *send_buffer;
322
323 size_t size_type;
324 MPI_Datatype mpi_type = get_MPI_number_type<T>(size_type);
325
326 if (from_node.rank != hila::myrank() && boundary_need_to_communicate(d)) {
327
328 // HANDLE RECEIVES: get node which will send here
329
330 // buffer can be separate or in Field buffer
331 receive_buffer = fs->get_receive_buffer(d, par, from_node);
332
333 size_t n = from_node.n_sites(par) * size / size_type;
334
335 if (n >= (1ULL << 31)) {
336 hila::out << "Too large MPI message! Size " << n << '\n';
337 hila::terminate(1);
338 }
339
340 post_receive_timer.start();
341
342 // c++ version does not return errors
343 MPI_Irecv(receive_buffer, (int)n, mpi_type, from_node.rank, tag, lattice.mpi_comm_lat,
344 &fs->receive_request[par_i][d]);
345
346 post_receive_timer.stop();
347 }
348
349 if (to_node.rank != hila::myrank() && boundary_need_to_communicate(-d)) {
350 // HANDLE SENDS: Copy Field elements on the boundary to a send buffer and send
351
352 unsigned sites = to_node.n_sites(par);
353
354 if (fs->send_buffer[d] == nullptr)
355 fs->send_buffer[d] = fs->payload.allocate_mpi_buffer(to_node.sites);
356
357 send_buffer = fs->send_buffer[d] + to_node.offset(par);
358
359 fs->gather_comm_elements(d, par, send_buffer, to_node);
360
361 size_t n = sites * size / size_type;
362#ifdef GPU_AWARE_MPI
363 gpuStreamSynchronize(0);
364 // gpuDeviceSynchronize();
365#endif
366
367 start_send_timer.start();
368
369 MPI_Isend(send_buffer, (int)n, mpi_type, to_node.rank, tag, lattice.mpi_comm_lat,
370 &fs->send_request[par_i][d]);
371
372 start_send_timer.stop();
373 }
374
375 // and do the boundary shuffle here, after MPI has started
376 // NOTE: there should be no danger of MPI and shuffle overwriting, MPI writes
377 // to halo buffers only if no permutation is needed. With a permutation MPI
378 // uses special receive buffer
379 fs->set_local_boundary_elements(d, par);
380
381 return get_dir_mask(d);
382}

393template <typename T>
394void Field<T>::wait_gather(Direction d, Parity p) const {
395
396 lattice_struct::nn_comminfo_struct &ci = lattice.nn_comminfo[d];
397 lattice_struct::comm_node_struct &from_node = ci.from_node;
398 lattice_struct::comm_node_struct &to_node = ci.to_node;
399
400 // check if this is done - either gathered or no comm to be done in the 1st place
401 if (is_gathered(d, p))
402 return;
403
404 // this is the branch if no comms -- shuffle was done in start_gather
405 if (from_node.rank == hila::myrank() && to_node.rank == hila::myrank())
406 return;
407
408 // if (!is_gather_started(d,p)) {
409 // hila::out0 << "Wait gather error - wait_gather without corresponding
410 // start_gather\n"; exit(1);
411 // }
412
413 // Note: the move can be Parity p OR ALL -- need to wait for it in any case
414 // set par to be the "sum" over both parities
415 // There never should be ongoing ALL and other parity gather -- start_gather takes
416 // care
417
418 // check here consistency, this should never happen
419 if (p != ALL && is_gather_started(d, p) && is_gather_started(d, ALL)) {
420 exit(1);
421 }
422
423 Parity par;
424 int n_wait = 1;
425 // what par to wait for?
426 if (is_gather_started(d, p))
427 par = p; // standard match
428 else if (p != ALL) {
429 if (is_gather_started(d, ALL))
430 par = ALL; // if all is running wait for it
431 else {
432 exit(1);
433 }
434 } else {
435 // now p == ALL and ALL is not running
436 if (is_gathered(d, EVEN) && is_gather_started(d, ODD))
437 par = ODD;
438 else if (is_gathered(d, ODD) && is_gather_started(d, EVEN))
439 par = EVEN;
440 else if (is_gather_started(d, EVEN) && is_gather_started(d, ODD)) {
441 n_wait = 2; // need to wait for both!
442 par = ALL;
443 } else {
444 exit(1);
445 }
446 }
447
448 if (n_wait == 2)
449 par = EVEN; // we'll flip both
450
451 for (int wait_i = 0; wait_i < n_wait; ++wait_i) {
452
453 int par_i = (int)par - 1;
454
455 if (from_node.rank != hila::myrank() && boundary_need_to_communicate(d)) {
456 wait_receive_timer.start();
457
458 MPI_Status status;
459 MPI_Wait(&fs->receive_request[par_i][d], &status);
460
461 wait_receive_timer.stop();
462
463#ifndef VANILLA
464 fs->place_comm_elements(d, par, fs->get_receive_buffer(d, par, from_node), from_node);
465#endif
466 }
467
468 // then wait for the sends
469 if (to_node.rank != hila::myrank() && boundary_need_to_communicate(-d)) {
470 wait_send_timer.start();
471 MPI_Status status;
472 MPI_Wait(&fs->send_request[par_i][d], &status);
473 wait_send_timer.stop();
474 }
475
476 // Mark the parity gathered from Direction dir
477 mark_gathered(d, par);
478
479 // Keep count of communications
480 lattice.n_gather_done += 1;
481
482 par = opp_parity(par); // flip if 2 loops
483 }
484}

489template <typename T>
490void Field<T>::field_struct::gather_elements(T *RESTRICT buffer,
491 const std::vector<CoordinateVector> &coord_list,
492 int root) const {
493
494 std::vector<unsigned> index_list;
495 std::vector<int> sites_on_rank(lattice.n_nodes());
496 std::vector<int> reshuffle_list(coord_list.size());
497
498 std::fill(sites_on_rank.begin(), sites_on_rank.end(), 0);
499
500 int nranks = 0;
501
502 int i = 0;
503 for (const CoordinateVector &c : coord_list) {
504 int rank = lattice.node_rank(c);
505 if (hila::myrank() == rank) {
506 index_list.push_back(lattice.site_index(c));
507 }
508
509 if (sites_on_rank[rank] == 0 && rank != root)
510 nranks++;
511 sites_on_rank[rank]++;
512 reshuffle_list[i++] = rank;
513 }
514
515 std::vector<T> send_buffer(index_list.size());
516 payload.gather_elements((T *)send_buffer.data(), index_list.data(), send_buffer.size(),
517 lattice);
518 if (hila::myrank() != root && sites_on_rank[hila::myrank()] > 0) {
519 MPI_Send((char *)send_buffer.data(), sites_on_rank[hila::myrank()] * sizeof(T), MPI_BYTE,
520 root, hila::myrank(), lattice.mpi_comm_lat);
521 }
522 if (hila::myrank() == root) {
523
524 // allocate buffer for receiving data
525 T *b;
526 std::vector<T> pb(coord_list.size() - sites_on_rank[root]);
527 b = pb.data();
528 // vector for node ptrs -- point to stuff from nodes
529 std::vector<T *> nptr(lattice.n_nodes());
530
531 std::vector<MPI_Request> mpi_req(nranks);
532 int nreqs = 0;
533 for (int n = 0; n < sites_on_rank.size(); n++) {
534 if (sites_on_rank[n] > 0) {
535 if (n != root) {
536 MPI_Status status;
537 MPI_Irecv(b, (int)(sites_on_rank[n] * sizeof(T)), MPI_BYTE, n, n,
538 lattice.mpi_comm_lat, &mpi_req[nreqs++]);
539
540 nptr[n] = b;
541 b += sites_on_rank[n];
542
543 } else {
544
545 nptr[n] = send_buffer.data();
546 }
547 }
548 }
549
550 if (nreqs > 0) {
551 std::vector<MPI_Status> stat_arr(nreqs);
552 MPI_Waitall(nreqs, mpi_req.data(), stat_arr.data());
553 }
554
555 // copy the data from bp to buffer, reordering
556 for (int i = 0; i < coord_list.size(); i++) {
557 buffer[i] = *nptr[reshuffle_list[i]];
558 nptr[reshuffle_list[i]]++;
559 }
560 }
561}

566template <typename T>
567void Field<T>::field_struct::scatter_elements(T *RESTRICT buffer,
568 const std::vector<CoordinateVector> &coord_list,
569 int root) {
570
571 std::vector<unsigned> index_list;
572 std::vector<int> sites_on_rank(lattice.n_nodes());
573 std::vector<int> reshuffle_list(coord_list.size());
574 std::fill(sites_on_rank.begin(), sites_on_rank.end(), 0);
575
576 int nranks = 0;
577 int i = 0;
578 for (CoordinateVector c : coord_list) {
579 int rank = lattice.node_rank(c);
580 if (hila::myrank() == rank) {
581 index_list.push_back(lattice.site_index(c));
582 }
583
584 if (sites_on_rank[rank] == 0 && rank != root)
585 nranks++;
586 sites_on_rank[rank]++;
587 reshuffle_list[i++] = rank;
588 }
589
590 // payload.gather_elements((T *)recv_buffer.data(), index_list.data(),
591 // recv_buffer.size(), lattice);
592
593 if (hila::myrank() != root && sites_on_rank[hila::myrank()] > 0) {
594 std::vector<T> recv_buffer(index_list.size());
595 MPI_Status status;
596
597 MPI_Recv((char *)recv_buffer.data(), sites_on_rank[hila::myrank()] * sizeof(T), MPI_BYTE,
598 root, hila::myrank(), lattice.mpi_comm_lat, &status);
599
600 payload.place_elements((T *)recv_buffer.data(), index_list.data(), recv_buffer.size(),
601 lattice);
602 }
603 if (hila::myrank() == root) {
604 // reordering buffers
605 std::vector<T> pb(coord_list.size());
606 // vector for node counters -- point to stuff from nodes
607 std::vector<unsigned> nloc(lattice.n_nodes());
608 std::vector<unsigned> ncount(lattice.n_nodes());
609 nloc[0] = ncount[0] = 0;
610
611 for (int n = 1; n < lattice.n_nodes(); n++) {
612 nloc[n] = nloc[n - 1] + sites_on_rank[n - 1];
613 ncount[n] = 0;
614 }
615 for (int i = 0; i < coord_list.size(); i++) {
616 int node = reshuffle_list[i];
617 pb[nloc[node] + ncount[node]] = buffer[i];
618 ncount[node]++;
619 }
620
621 std::vector<MPI_Request> mpi_req(nranks);
622 int nreqs = 0;
623 for (int n = 0; n < sites_on_rank.size(); n++) {
624 if (sites_on_rank[n] > 0) {
625 if (n != root) {
626 MPI_Isend(pb.data() + nloc[n], (int)(sites_on_rank[n] * sizeof(T)), MPI_BYTE, n,
627 n, lattice.mpi_comm_lat, &mpi_req[nreqs++]);
628 }
629 }
630 }
631
632 payload.place_elements(pb.data() + nloc[root], index_list.data(), index_list.size(),
633 lattice);
634
635 if (nreqs > 0) {
636 std::vector<MPI_Status> stat_arr(nreqs);
637 MPI_Waitall(nreqs, mpi_req.data(), stat_arr.data());
638 }
639 }

678template <typename T>
679std::vector<T> Field<T>::get_subvolume(const CoordinateVector &cmin, const CoordinateVector &cmax,
680 bool bcast) const {
681
682 size_t vol = 1;
683 foralldir(d) {
684 vol *= cmax[d] - cmin[d] + 1;
685 assert(cmax[d] >= cmin[d] && cmin[d] >= 0 && cmax[d] < lattice.size(d));
686 }
687 std::vector<CoordinateVector> clist(vol);
688 CoordinateVector c;
689
690 size_t i = 0;
691 forcoordinaterange(c, cmin, cmax) {
692 clist[i++] = c;
693 }
694 return get_elements(clist, bcast);
695}

717template <typename T>
718void Field<T>::copy_local_data(std::vector<T> &buffer) const {
719
720 // copy to local variables to avoid lattice ptr
721 CoordinateVector nmin = lattice.mynode.min;
722 Vector<NDIM, unsigned> nmul = lattice.mynode.size_factor;
723
724 buffer.resize(lattice.mynode.volume());
725#if defined(CUDA) || defined(HIP)
726 // d_malloc mallocs from device if needed
727 T *data = (T *)d_malloc(sizeof(T) * lattice.mynode.volume());
728#else
729 T *data = buffer.data();
730#endif
731
732#pragma hila novector direct_access(data)
733 onsites(ALL) {
734 Vector<NDIM, unsigned> nodec;
735 nodec = X.coordinates() - nmin;

787template <typename T>
788inline void collect_field_halo_data_(T *data, const Field<T> &src, Field<T> &dest,
789 const Vector<NDIM, int> &dirs, int ndir) {
790
791 // get the coords of the min point of the halo array
792 CoordinateVector nmin = lattice.mynode.min;
793 nmin.asArray() -= 1;
794
795 // construct the mult vector to access the data
796 Vector<NDIM, unsigned> nmul;
797 nmul.e(0) = 1;
798 for (int i = 1; i < NDIM; i++)
799 nmul.e(i) = nmul.e(i - 1) * (lattice.mynode.size[i - 1] + 2);
800
801
802 Vector<NDIM, int> node_min;
803 Vector<NDIM, int> node_max;
804 foralldir(d) {
805 node_min[d] = lattice.mynode.min[d];
806 node_max[d] = lattice.mynode.min[d] + lattice.mynode.size[d] - 1;
807 }
808
809#pragma hila novector direct_access(data)
810 onsites(ALL) {
811 Vector<NDIM, unsigned> nodec;
812 CoordinateVector c = X.coordinates();
813 bool gotit = true;
814
815 for (int i = 0; i < ndir; i++) {
816 Direction d = (Direction)dirs[i];
817 if (c.e(d) == node_min[d]) {
818 c.e(d) -= 1;
819 } else if (c.e(d) == node_max[d]) {
820 c.e(d) += 1;
821 } else
822 gotit = false;
823 }
824
825 Direction d = (Direction)dirs[ndir];
826 if (gotit && c.e(d) == node_min[d]) {
827 c.e(d) -= 1;
828 nodec = c - nmin;
829 data[nodec.dot(nmul)] = src[X - d];
830 dest[X] = src[X - d];
831
832 } else if (gotit && c.e(d) == node_max[d]) {
833 c.e(d) += 1;
834 nodec = c - nmin;
835 data[nodec.dot(nmul)] = src[X + d];
836 dest[X] = src[X + d];
837 }
838 }
839}
